# Food auto-build triggers — eligibility, affordability, pop, and cap checks.
# Gold threshold checks are done in the effects file (per-building script values).

# Root is the location. Expects $target$ = a building_type. Saves target as scope:building_type.
# Combines the four-check cluster shared by every food option: can-build (engine prerequisites),
# not-at-max-level-and-not-already-under-construction, save-scope-for-affordability-macros,
# and construction-goods affordability.
# For 1-level buildings (e.g. windmill), the max-level check also subsumes NOT has_building.
lsq_food_location_buildable_for_building_type = {
	lsq_location_allowed_to_build = { target = $target$ }
	lsq_location_building_type_is_not_at_max_level = { target = $target$ }
	$target$ = { save_temporary_scope_as = building_type }
	lsq_can_afford_switch_based_on_building_type = yes
}

# ─── Location matching ───

# Root is the location. Expects scope:food_good (the goods scope from the CMM list).
# True if this location's RGO produces the specified food good.
lsq_food_location_matches_rgo_type = {
	raw_material = scope:food_good
}

# Root is the location. Expects scope:food_good and scope:country.
# True if location passes the river_only filter for this food good.
lsq_food_location_meets_river_filter = {
	trigger_if = { limit = { scope:country = { is_target_in_variable_list = { name = lsq_auto_food_rgo_types_river_only target = scope:food_good } } }
		OR = { has_river = yes is_adjacent_to_lake = yes }
	} trigger_else = {
		always = yes
	}
}

# ─── Combined pool pop check ───

# Root is the location. Expects scope:country.
# True when ANY pop type in the location is scarce — existing buildings/RGO haven't filled their
# workers yet and promotion isn't fast enough. When true, the food system should NOT build anything
# new — let pops promote and fill existing positions first. This treats all pop types as a combined
# pool: peasants promote to laborers, so building a peasant-consuming building (farming village)
# when laborers are scarce diverts peasants from promoting to fill the RGO.
lsq_food_location_pops_still_filling = {
	OR = {
		NOT = { lsq_check_location_has_laborers_available_or_will_be = yes }
		NOT = { lsq_check_location_has_peasants_available_or_will_be = yes }
	}
}

# ─── Can-build checks per option ───
# Each checks: requirements met, not at cap, construction goods available, pops available.
# Gold threshold is checked separately in the effects file.
# Root is the location. Expects scopes: country, location.

# Option 1: +1 RGO level
lsq_food_can_build_rgo = {
	AND = {
		lsq_is_rgo_not_max_level_already = yes
		lsq_is_rgo_under_construction = no
		lsq_has_enough_raw_materials_for_rgo = yes
		lsq_check_location_has_laborers_available_or_will_be = yes
	}
}

# Option 2: +1 Farming Village level
# Requires: farming RGO, rural settlement, not at max level, not already under construction
lsq_food_can_build_farming_village = {
	AND = {
		location_rank = location_rank:rural_settlement
		OR = {
			raw_material = goods:wheat
			raw_material = goods:maize
			raw_material = goods:rice
			raw_material = goods:millet
			raw_material = goods:legumes
			raw_material = goods:potato
			raw_material = goods:livestock
			raw_material = goods:olives
			raw_material = goods:fruit
			raw_material = goods:wool
		}
		lsq_food_location_buildable_for_building_type = { target = building_type:farming_village }
		lsq_check_location_has_peasants_available_or_will_be = yes
	}
}

# Option 3: Windmill (1 level max)
# Requires: wheat/rice/millet RGO, rural, windmills advance researched
lsq_food_can_build_windmill = {
	AND = {
		location_rank = location_rank:rural_settlement
		OR = {
			raw_material = goods:wheat
			raw_material = goods:rice
			raw_material = goods:millet
		}
		scope:country = { has_advance = windmills_advance }
		lsq_food_location_buildable_for_building_type = { target = building_type:windmill }
		lsq_check_location_has_laborers_available_or_will_be = yes
	}
}

# Option 4: +1 Irrigation level
# Requires: river OR adjacent to lake, not at max level, not already under construction
lsq_food_can_build_irrigation = {
	AND = {
		OR = { has_river = yes is_adjacent_to_lake = yes }
		lsq_food_location_buildable_for_building_type = { target = building_type:irrigation_systems }
		lsq_check_location_has_laborers_available_or_will_be = yes
	}
}

# Option 5: +1 Fishing Village level
# Requires: coastal, rural, not at max level, not already under construction
lsq_food_can_build_fishing_village = {
	AND = {
		is_coastal = yes
		location_rank = location_rank:rural_settlement
		lsq_food_location_buildable_for_building_type = { target = building_type:fishing_village }
		lsq_check_location_has_peasants_available_or_will_be = yes
	}
}

# Option 6: +1 Sheep Farms level
# Requires: sparse/farmland/grasslands vegetation, rural, not at max level, not already under construction
lsq_food_can_build_sheep_farms = {
	AND = {
		location_rank = location_rank:rural_settlement
		OR = {
			vegetation = sparse
			vegetation = farmland
			vegetation = grasslands
		}
		lsq_food_location_buildable_for_building_type = { target = building_type:sheep_farms }
		lsq_check_location_has_peasants_available_or_will_be = yes
	}
}

# Option 7: +1 Forest Village level
# Requires: wild_game RGO OR forest/woods/jungle vegetation, rural, not at max level, not already under construction
lsq_food_can_build_forest_village = {
	AND = {
		location_rank = location_rank:rural_settlement
		OR = {
			raw_material = goods:wild_game
			vegetation = forest
			vegetation = woods
			vegetation = jungle
		}
		lsq_food_location_buildable_for_building_type = { target = building_type:forest_village }
		lsq_check_location_has_laborers_available_or_will_be = yes
	}
}

# ─── Per-option viability (can-build AND affordable AND meets minimum discount) ───
# These combine the three gates the effects file applies before constructing each option.
# Root is the location. Expects scopes: country, location.

lsq_food_option_rgo_viable = {
	lsq_food_can_build_rgo = yes
	lsq_check_country_gold_above = { threshold = lsq_food_expected_gold_for_rgo }
	lsq_rgo_meets_minimum_discount = { threshold = lsq_get_min_discount_for_auto_food }
}

lsq_food_option_farming_village_viable = {

# True if ANY of the seven food-building options is currently viable at this location.
# Used by the location auto-food button to decide green vs red glow.
lsq_food_location_has_buildable_option = {
	OR = {
		lsq_food_option_rgo_viable = yes
		lsq_food_option_farming_village_viable = yes
		lsq_food_option_windmill_viable = yes
		lsq_food_option_irrigation_viable = yes
		lsq_food_option_fishing_village_viable = yes
		lsq_food_option_sheep_farms_viable = yes
		lsq_food_option_forest_village_viable = yes
	}
}
