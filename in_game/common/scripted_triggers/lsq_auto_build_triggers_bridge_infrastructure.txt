# Root is the location, also expects the scope country
lsq_location_will_auto_build_bridge_infrastructure = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_bridge_infrastructure = yes
		lsq_auto_build_check_meets_minimum_control_bridge_infrastructure = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_bridge_infrastructure = yes
		lsq_auto_build_check_pops_available_bridge_infrastructure = yes
		lsq_auto_build_check_can_afford_bridge_infrastructure = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_bridge_infrastructure = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:bridge_infrastructure }
}

# Root is the location
lsq_auto_build_check_meets_minimum_market_price_impact_bridge_infrastructure = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_market_price_impact_with_key = { key = bridge_infrastructure }
}
