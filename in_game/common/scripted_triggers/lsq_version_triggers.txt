# Expects scopes: country
lsq_is_version_set = {
  AND = {
    exists = scope:country.var:lsq_major_version
    exists = scope:country.var:lsq_minor_version
    exists = scope:country.var:lsq_patch_version
  }
}

# Expects scopes: country
lsq_is_version = {
  AND = {
    scope:country.var:lsq_major_version = $major$
    scope:country.var:lsq_minor_version = $minor$
    scope:country.var:lsq_patch_version = $patch$
  }
}
