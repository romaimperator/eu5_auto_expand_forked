lsq_is_version_set = {
  AND = {
    exists = global_var:lsq_major_version
    exists = global_var:lsq_minor_version
    exists = global_var:lsq_patch_version
  }
}

lsq_is_version = {
  AND = {
    global_var:lsq_major_version = $major$
    global_var:lsq_minor_version = $minor$
    global_var:lsq_patch_version = $patch$
  }
}
