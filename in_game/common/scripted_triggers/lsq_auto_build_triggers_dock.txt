# Root is the location, also expects the scope country
lsq_location_can_auto_build_dock = {
	AND = {
		lsq_location_will_auto_build_dock = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_dock = {
	AND = {
		always = yes # not locked behind an advance
		NOT = { has_building = building_type:dock }
		is_port = yes
		OR = {
			location_rank = location_rank:town
			location_rank = location_rank:city
		}
	}
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_dock = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_dock = yes
		lsq_auto_build_check_meets_minimum_control_dock = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_dock = yes
		lsq_auto_build_check_pops_available_dock = yes
		lsq_auto_build_check_can_afford_dock = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location

	lsq_auto_build_check_meets_minimum_control = { target = building_type:dock }
}

	lsq_auto_build_check_meets_minimum_market_price_impact_with_key = { key = dock }
