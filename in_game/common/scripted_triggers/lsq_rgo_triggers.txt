# This trigger is used by the GUI to see if the prerequisites for auto expanding RGO are met
lsq_is_able_to_auto_expand_rgo = {
	AND = {
		owner = { lsq_should_pause_auto_expand = no }
		lsq_is_rgo_not_max_level_already = yes
		lsq_is_rgo_fully_employed = yes
		# lsq_has_enough_raw_materials_for_rgo = yes
	}
}

# This trigger is used by the on action to determine if the RGO should be auto expanded
# Expects scopes: country, location
lsq_should_rgo_auto_expand = {
	scope:location = {
		AND = {
			scope:country = { lsq_should_pause_auto_expand = no }
			lsq_is_rgo_not_max_level_already = yes
			lsq_is_rgo_fully_employed = yes
			# lsq_has_enough_raw_materials_for_rgo = yes
			lsq_is_rgo_under_construction = no
			lsq_is_there_enough_gold_available = yes
		}
	}
}

lsq_is_rgo_not_max_level_already = {
	lsq_modifier_rgo_level < lsq_get_max_rgo_workers_floor
}

lsq_is_rgo_fully_employed = {
	lsq_auto_build_check_laborers_available_or_will_be = yes
}

lsq_is_rgo_under_construction = {
	lsq_get_rgos_under_construction > 0
}

# Expects scopes: country, location
lsq_is_there_enough_gold_available = {
	scope:country.gold > lsq_expected_gold_available_for_rgo_auto_expand
}
