# Root is the location, also expects the scope country
lsq_location_can_auto_build_sawmill = {
	AND = {
		lsq_location_will_auto_build_sawmill = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_sawmill = {
	AND = {
		lsq_auto_build_check_building_researched_sawmill = yes
		lsq_auto_build_check_no_building_type = { target = building_type:sawmill }
		lsq_auto_build_check_rural = yes
		OR = {
			vegetation = woods
			vegetation = forest
			vegetation = farmland
			vegetation = jungle
			raw_material = goods:lumber
		}
	}
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_sawmill = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_sawmill = yes
		lsq_auto_build_check_meets_minimum_control_sawmill = yes
		lsq_auto_build_check_pops_available_sawmill = yes
		lsq_auto_build_check_rgo_level_requirement_met_sawmill = yes
		lsq_auto_build_check_can_afford_sawmill = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_sawmill = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:sawmill }
}

# Root is the location
lsq_auto_build_check_pops_available_sawmill = {
	lsq_auto_build_check_laborers_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_sawmill = {
	lsq_can_afford_sawmill = yes
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_sawmill = {
	always = yes # not locked behind an advance
}

# Root is the location, also expects the scope country
lsq_auto_build_check_rgo_level_requirement_met_sawmill = {
	OR = {
		AND = { raw_material = goods:lumber lsq_modifier_rgo_level >= scope:country.var:lsq_required_rgo_levels_before_sawmill }
		AND = { has_building = building_type:lumber_mill "building(building_type:lumber_mill|owner)" = { building_level >= scope:country.var:lsq_required_rgo_levels_before_sawmill } }
		scope:country.var:lsq_required_rgo_levels_before_sawmill = 0
	}
}
