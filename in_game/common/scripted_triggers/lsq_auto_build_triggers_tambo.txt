# Root is the location, also expects the scope country
lsq_location_can_auto_build_tambo = {
	AND = {
		lsq_location_will_auto_build_tambo = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_tambo = {
	AND = {
		scope:country = { has_advance = tambo_advance }
		NOT = { has_building = building_type:tambo }
		lsq_auto_build_check_has_at_least_one_road = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_tambo = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_tambo = yes
		lsq_auto_build_check_meets_minimum_control_tambo = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_tambo = yes
		lsq_auto_build_check_pops_available_tambo = yes
		lsq_auto_build_check_can_afford_tambo = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_tambo = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:tambo }
}

# Root is the location
lsq_auto_build_check_meets_minimum_market_price_impact_tambo = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_market_price_impact_with_key = { key = tambo }
}

# Root is the location
lsq_auto_build_check_pops_available_tambo = {
	lsq_auto_build_check_soldiers_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_tambo = {
	lsq_can_afford_switch_based_on_building_type = { target = building_type:tambo }
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_tambo = {
	scope:country = { has_advance = tambo_advance }
}
