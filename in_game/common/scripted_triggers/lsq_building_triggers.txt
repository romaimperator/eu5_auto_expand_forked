# Root is the building. Checks that building potential profit (adjusted for control) meets the minimum threshold.
# If lsq_minimum_profit_for_buildings is 0 or not set, the check always passes.
lsq_building_meets_minimum_profit = {
	OR = {
		NOT = { owner = { exists = var:lsq_minimum_profit_for_buildings } }
		owner.var:lsq_minimum_profit_for_buildings = 0
		lsq_building_potential_profit_after_control >= owner.var:lsq_minimum_profit_for_buildings
	}
}

# Root is the building
lsq_building_can_auto_expand_other_than_checking_for_upgrades = {
	location = { save_temporary_scope_as = location }
	building_type = { save_temporary_scope_as = building_type }
	AND = {
		OR = {
			AND = { building_type = { lsq_is_building_type_production_building = yes } lsq_is_building_profitable = yes lsq_building_meets_minimum_profit = yes }
			building_type = { lsq_is_building_type_non_production_multi_level_building = yes }
		}
		lsq_is_building_open = yes
		lsq_is_building_fully_employed = yes
		lsq_is_building_lacking_goods = no
		location = { lsq_location_has_enough_pops_available = yes }
		location = { lsq_is_location_under_building_limit = yes }
		location = { can_build_building = scope:building_type }
	}
}

# Root is the building. Same as lsq_building_can_auto_expand_other_than_checking_for_upgrades but without

	location = { save_temporary_scope_as = location }

	AND = {
		OR = {
			AND = { building_type = { lsq_is_building_type_production_building = yes } lsq_is_building_profitable = yes lsq_building_meets_minimum_profit = yes }
			building_type = { lsq_is_building_type_non_production_multi_level_building = yes }
		}
		lsq_is_building_open = yes
		lsq_is_building_fully_employed = yes
		lsq_is_building_lacking_goods = no
		location = { lsq_location_has_enough_pops_available = yes }
		location = { lsq_is_location_under_building_limit = yes }
	}
}
