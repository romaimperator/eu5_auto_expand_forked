# Root is the building and requires local_var:double_building_cost
lsq_is_there_enough_gold_available_for_building = {
	owner = {
		OR = {
			AND = { exists = var:lsq_minimum_gold_for_buildings gold > { value = local_var:double_building_cost add = var:lsq_minimum_gold_for_buildings } }
			AND = { NOT = { exists = var:lsq_minimum_gold_for_buildings } gold > local_var:double_building_cost }
		}
	}
}

# Root is the building
lsq_is_building_obsolete = {
	building_type = { save_temporary_scope_as = the_building_type }
	owner = { building_type_is_obsolete = scope:the_building_type }
}

# Root is the building
lsq_is_building_upgradeable = {
	building_type = { is_upgradeable = yes }
}

# Root is the building
lsq_building_can_be_upgraded_by_owner = {
	building_can_be_upgraded_by = owner
}

# Root is the building
lsq_is_building_upgradeable_and_can_be_upgraded_by_owner = {
	AND = {
		lsq_is_building_upgradeable = yes
		lsq_building_can_be_upgraded_by_owner = yes
		owner = { lsq_should_auto_expand_upgrade_buildings = yes }
	}
}

# Root is the building
lsq_is_building_not_max_level_and_not_obsolete_or_is_building_type_upgradeable_and_can_be_upgraded_by_owner = {
	OR = {
		lsq_is_building_not_max_level_already_and_not_obsolete = yes
		lsq_is_building_upgradeable_and_can_be_upgraded_by_owner = yes
	}
}

# Root is the building
lsq_is_building_lacking_goods = {

}

# Root is the building
lsq_is_location_under_building_limit = {
	location = { total_building_levels < modifier:free_building_levels }
}

# Root is the building
lsq_is_building_at_max_level = {
	is_at_max_level = yes
}

# Root is the building
lsq_is_building_not_max_level_already_and_not_obsolete = {
	AND = {
		lsq_is_building_at_max_level = no
		lsq_is_building_obsolete = no
	}
}

# Root is the building
lsq_is_location_pops_available = {
	AND = {
		NAND = { building_pop_type = burghers location = { is_burgher_positive_deficit = no } }  # checks if promoted pops of type are available if building type employes that pop type
		NAND = { building_pop_type = laborers location = { is_labourer_positive_deficit = no } } # same here :3
	}
}

# Root is the building
lsq_is_building_fully_employed = {
	AND = {
		building_potential_profit = building_profit
		OR = {
			is_full_capacity = yes
			building_level = 0
		}
	}
}

# Root is the building
lsq_is_building_open = {
	is_opened = yes
}

# Root is the building
lsq_is_building_profitable = {
	OR = {
		is_profitable = yes
		building_level = 0
	}
}

# Root is the building
lsq_building_can_auto_expand_other_than_checking_for_upgrades = {
	AND = {
		lsq_is_building_profitable = yes
		lsq_is_building_open = yes
		lsq_is_building_fully_employed = yes
		lsq_is_not_lacking_goods = yes
		lsq_is_location_pops_available = yes
		lsq_is_location_under_building_limit = yes
	}
}

lsq_is_not_under_construction_and_gold_is_available = {

}
