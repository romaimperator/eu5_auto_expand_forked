# Root is the location, also expects the scope country
lsq_location_can_auto_build_library = {
	AND = {
		lsq_location_will_auto_build_library = yes
		building_type:library = { lsq_auto_build_check_owner_gold_available = yes }
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_library = {
	AND = {
		lsq_auto_build_check_building_researched_library = yes
		lsq_auto_build_check_no_building_type = { target = building_type:library }
		lsq_auto_build_check_town_or_city = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_library = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_library = yes
		lsq_auto_build_check_pops_available_library = yes
	}
}

# Root is the location
lsq_auto_build_check_pops_available_library = {
	lsq_auto_build_check_burghers_available_or_will_be = yes
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_library = {
	scope:country = { has_advance = library_advance }
}
