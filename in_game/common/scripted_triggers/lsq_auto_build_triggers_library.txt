# Root is the location, also expects the scope country
lsq_location_can_auto_build_library = {
	AND = {
		lsq_location_will_auto_build_library = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_library = {
	AND = {
		scope:country = { has_advance = library_advance }
		NOT = { has_building = building_type:library }
		OR = {
			location_rank = location_rank:town
			location_rank = location_rank:city
		}
	}
	# AND = {
	# 	lsq_auto_build_check_building_researched_library = yes
	# 	lsq_auto_build_check_no_building_type = { target = building_type:library }
	# 	lsq_auto_build_check_town_or_city = yes
	# }
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_library = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_library = yes
		lsq_auto_build_check_meets_minimum_control_library = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_library = yes
		lsq_auto_build_check_pops_available_library = yes
		lsq_auto_build_check_can_afford_library = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_library = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:library }
}

# Root is the location
lsq_auto_build_check_meets_minimum_market_price_impact_library = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_market_price_impact_with_key = { key = library }
}

# Root is the location
lsq_auto_build_check_pops_available_library = {
	lsq_auto_build_check_burghers_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_library = {
	building_type:library = { save_temporary_scope_as = building_type } lsq_can_afford_switch_based_on_building_type = yes
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_library = {
	scope:country = { has_advance = library_advance }
}
