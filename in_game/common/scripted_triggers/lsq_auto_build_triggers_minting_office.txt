# Root is the location, also expects the scope country
lsq_location_can_auto_build_minting_office = {
	AND = {
		lsq_location_will_auto_build_minting_office = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_minting_office = {
	AND = {
		scope:country = { has_advance = mint_advance }
		NOT = { has_building = building_type:minting_office }
		OR = {
			location_rank = location_rank:town
			location_rank = location_rank:city
		}
	}
	# AND = {
	# 	lsq_auto_build_check_building_researched_minting_office = yes
	# 	lsq_auto_build_check_no_building_type = { target = building_type:minting_office }
	# 	lsq_auto_build_check_town_or_city = yes
	# }
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_minting_office = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_minting_office = yes
		lsq_auto_build_check_meets_minimum_control_minting_office = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_minting_office = yes
		lsq_auto_build_check_pops_available_minting_office = yes
		lsq_auto_build_check_can_afford_minting_office = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_minting_office = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:minting_office }
}

# Root is the location
lsq_auto_build_check_meets_minimum_market_price_impact_minting_office = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_market_price_impact_with_key = { key = minting_office }
}

# Root is the location
lsq_auto_build_check_pops_available_minting_office = {
	lsq_auto_build_check_burghers_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_minting_office = {
	lsq_can_afford_minting_office = yes
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_minting_office = {
	scope:country = { has_advance = mint_advance }
}
