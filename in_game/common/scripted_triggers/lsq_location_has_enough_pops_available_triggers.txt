# Root is the location, also expects scope:building_type and the scope country
lsq_location_has_enough_pops_available = {
	has_global_variable_map = lsq_building_type_to_pop_type
	switch = { trigger = "global_variable_map(lsq_building_type_to_pop_type|scope:building_type)"
		pop_type:nobles = { lsq_check_location_has_nobles_available_or_will_be = yes }
		pop_type:clergy = { lsq_check_location_has_clergy_available_or_will_be = yes }
		pop_type:burghers = { lsq_check_location_has_burghers_available_or_will_be = yes }
		pop_type:laborers = { lsq_check_location_has_laborers_available_or_will_be = yes }
		pop_type:soldiers = { lsq_check_location_has_soldiers_available_or_will_be = yes }
		pop_type:peasants = { lsq_check_location_has_peasants_available_or_will_be = yes }
		pop_type:slaves = { lsq_check_location_has_slaves_available_or_will_be = yes }
		pop_type:tribesmen = { lsq_check_location_has_tribesmen_available_or_will_be = yes }
		fallback = { always = no }
	}
}
