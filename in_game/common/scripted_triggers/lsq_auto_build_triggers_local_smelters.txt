# Root is the location, also expects the scope country
lsq_location_can_auto_build_local_smelters = {
	AND = {
		lsq_location_will_auto_build_local_smelters = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_local_smelters = {
	AND = {
		scope:country = { has_advance = local_smelters_advance }
		NOT = { has_building = building_type:local_smelters }
		location_rank = location_rank:rural_settlement
		OR = {
			raw_material = goods:iron # 425
			raw_material = goods:copper # 223
			raw_material = goods:lead # 104
			raw_material = goods:tin # 91

# Root is the location, also expects the scope country
lsq_location_will_auto_build_local_smelters = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_local_smelters = yes
		lsq_auto_build_check_meets_minimum_control_local_smelters = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_local_smelters = yes
		lsq_auto_build_check_pops_available_local_smelters = yes
		lsq_auto_build_check_rgo_level_requirement_met_local_smelters = yes
		lsq_auto_build_check_can_afford_local_smelters = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_local_smelters = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:local_smelters }
}

# Root is the location
lsq_auto_build_check_meets_minimum_market_price_impact_local_smelters = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_market_price_impact_with_key = { key = local_smelters }
}

# Root is the location
lsq_auto_build_check_pops_available_local_smelters = {
	lsq_auto_build_check_laborers_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_local_smelters = {
	lsq_can_afford_local_smelters = yes
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_local_smelters = {
	scope:country = { has_advance = local_smelters_advance }
}

# Root is the location, also expects the scope country
lsq_auto_build_check_rgo_level_requirement_met_local_smelters = {
	AND = {
		OR = {
			raw_material = goods:iron
			raw_material = goods:copper
			raw_material = goods:tin
			raw_material = goods:lead
		}
		lsq_modifier_rgo_level >= "scope:country.variable_map(lsq_min_rgos_by_building_type|building_type:local_smelters)"
	}
}
