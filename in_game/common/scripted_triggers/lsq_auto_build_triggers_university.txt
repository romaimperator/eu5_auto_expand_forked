# Root is the location, also expects the scope country
lsq_location_can_auto_build_university = {
	AND = {
		lsq_location_will_auto_build_university = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_university = {
	AND = {
		scope:country = { has_advance = university_advance }
		NOT = { has_building = building_type:university }
		OR = {
			location_rank = location_rank:town
			location_rank = location_rank:city
		}
		is_core_of = owner
		owner = { modifier:university_construction_blocked = no }
	}
	# AND = {
	# 	lsq_auto_build_check_building_researched_university = yes
	# 	lsq_auto_build_check_no_building_type = { target = building_type:university }
	# 	lsq_auto_build_check_town_or_city = yes
	# 	lsq_auto_build_check_is_a_core = yes
	# 	lsq_auto_build_check_university_not_blocked = yes
	# }
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_university = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_university = yes
		lsq_auto_build_check_meets_minimum_control_university = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_university = yes
		lsq_auto_build_check_pops_available_university = yes
		lsq_auto_build_check_can_afford_university = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_university = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:university }
}

# Root is the location
lsq_auto_build_check_meets_minimum_market_price_impact_university = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_market_price_impact_with_key = { key = university }
}

# Root is the location
lsq_auto_build_check_pops_available_university = {
	lsq_auto_build_check_clergy_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_university = {
	lsq_can_afford_switch_based_on_building_type = { target = building_type:university }
}

# Root is the location, checking owner because we're assuming the modifier blocks construction in a subject if the subject has it but regardless of if the player has it since the game config reads like it works that way
lsq_auto_build_check_university_not_blocked = {
	owner = { modifier:university_construction_blocked = no }
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_university = {
	scope:country = { has_advance = university_advance }
}
