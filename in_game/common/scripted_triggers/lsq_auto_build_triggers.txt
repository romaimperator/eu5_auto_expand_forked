# Generic triggers for auto-building

# Shared gold-threshold check. Expects scope:country. Callers supply the script value to compare against.
lsq_check_country_gold_above = {
	scope:country.gold > $threshold$
}

# Root is the location. Expects $target$ = a building_type.
# True if the building type has no instance here OR the existing instance isn't at its max level
# AND has no level already under construction. Use for multi-level buildings to avoid queuing
# construction beyond the cap or double-queueing while a level is still being built.
lsq_location_building_type_is_not_at_max_level = {
	$target$ = { save_temporary_scope_as = target_building_type_for_max_check }
	trigger_if = { limit = { has_building = $target$ }
		"building(scope:target_building_type_for_max_check|owner)" = {
			is_at_max_level = no
			building_levels_under_construction = 0
		}
	}
}

# Root is the location
lsq_check_location_meets_min_market_access = {
	market_access > 0
}

# Root is the location. Expects the building type as target and scope country. True when location has enough RGO/lumber levels for this building type (sawmill: RGO or lumber_mill >= required; windmill/local_smelters: RGO >= required).
# Non-RGO buildings always pass (fallback = yes).
lsq_auto_build_check_rgo_level_requirement_met = {
	switch = { trigger = $target$
		building_type:sawmill = { lsq_auto_build_check_rgo_level_requirement_met_sawmill = yes }
		building_type:windmill = { lsq_auto_build_check_rgo_level_requirement_met_windmill = yes }
		building_type:local_smelters = { lsq_auto_build_check_rgo_level_requirement_met_local_smelters = yes }
		fallback = { always = yes }
	}
}

# Root is the location, also expects the scope country
lsq_auto_build_check_rgo_level_requirement_met_sawmill = {
	OR = {
		AND = { raw_material = goods:lumber lsq_modifier_rgo_level >= "scope:country.variable_map(lsq_min_rgos_by_building_type|building_type:sawmill)" }
		AND = { has_building = building_type:lumber_mill "building(building_type:lumber_mill|owner)" = { building_level >= "scope:country.variable_map(lsq_min_rgos_by_building_type|building_type:sawmill)" } }
		"scope:country.variable_map(lsq_min_rgos_by_building_type|building_type:sawmill)" = 0
	}
}

# Root is the location, also expects the scope country
lsq_auto_build_check_rgo_level_requirement_met_windmill = {
	AND = {
		OR = {
			raw_material = goods:wheat
			raw_material = goods:millet
			raw_material = goods:rice
		}
		lsq_modifier_rgo_level >= "scope:country.variable_map(lsq_min_rgos_by_building_type|building_type:windmill)"
	}
}

# Root is the location, also expects the scope country
lsq_auto_build_check_rgo_level_requirement_met_local_smelters = {
	AND = {
		OR = {
			raw_material = goods:iron
			raw_material = goods:copper
			raw_material = goods:tin
			raw_material = goods:lead
		}
		lsq_modifier_rgo_level >= "scope:country.variable_map(lsq_min_rgos_by_building_type|building_type:local_smelters)"
	}
}

# Expects the scope building_type. True only for sawmill, windmill, local_smelters (so tooltip row "Has required RGO levels" is shown only for those).
lsq_auto_build_show_rgo_level_tooltip_row = {
	OR = {
		scope:building_type = building_type:sawmill
		scope:building_type = building_type:windmill
		scope:building_type = building_type:local_smelters
	}
}

# Root is the country. Expects the building type as a target, also expects the scope country
lsq_auto_build_check_any_location_allowed_to_build = {
	trigger_if = { limit = { is_target_in_variable_list = { name = lsq_auto_build_building_types_in_subjects_enabled target = scope:building_type } }
		OR = {
			any_owned_location = {
				lsq_location_allowed_to_build = { target = $target$ }
			}
			any_subject_or_below = {
				any_owned_location = {
					lsq_location_allowed_to_build = { target = $target$ }
				}
			}
		}
	} trigger_else = {
		any_owned_location = {
			lsq_location_allowed_to_build = { target = $target$ }
		}
	}
}

# ─── Generic auto-build trigger chain ───
# These replace the old per-building trigger files. All building-specific
# logic is resolved dynamically via engine triggers and precomputed maps.

# Expects scopes: building_type, country. Root is the location.
lsq_location_can_auto_build = {
	lsq_auto_build_will_auto_build = { target = scope:building_type }
	lsq_check_country_gold_above = { threshold = lsq_expected_gold_available_for_building_type_for_auto_build }
}

# Root is the location. Expects the building type as a target, also expects the scope country.
lsq_auto_build_will_auto_build = {
	scope:country = { lsq_should_pause_auto_expand = no }
	save_temporary_scope_as = location
	# Auto-build only constructs one level per location. Multi-level upgrades
	# are handled by the auto-expand-buildings feature.
	NOT = { has_building = $target$ }
	lsq_location_allowed_to_build = { target = $target$ }
	lsq_auto_build_check_meets_minimum_control = { target = $target$ }
	lsq_auto_build_check_meets_minimum_market_price_impact = { target = $target$ }
	lsq_location_has_enough_pops_available = yes
	lsq_auto_build_check_rgo_level_requirement_met = { target = $target$ }
	lsq_check_location_meets_min_market_access = yes
}

# Root is the location. Expects the building type as a target, also expects the scope country.
# Uses can_build_building to check all placement prerequisites (tech, location rank, terrain, culture, etc.).
# Foreign-only building types (is_foreign = yes, e.g. hanseatic_kontor, fondaco) are excluded:
# can_build_building does not filter them out, but we do not support auto-expanding them in owned locations.
lsq_location_allowed_to_build = {
	scope:country = { save_temporary_scope_as = actor }
	$target$ = { is_foreign = no }
	can_build_building = $target$
	scope:country = { can_build_building = $target$ }
}

# Expects scopes: country, location
# Expects param: target => building_type
lsq_auto_build_check_meets_minimum_control = {
	scope:location.max_control >= $target$.lsq_get_minimum_control_for_auto_build
}

# Expects scopes: country, location
# Expects param: target => building_type
# The 0.005 added is a 0.5% tolerance so that a setting of -33% matches a cost
# adjustment of -32.5% (avoids float-rounding misses at the boundary).
lsq_auto_build_check_meets_minimum_market_price_impact = {
	OR = {
		$target$ = { save_temporary_scope_as = building_type lsq_get_standard_construction_demand_total = 0 }
		$target$ = { lsq_get_standard_construction_cost_adjustment_for_building_type <= { value = lsq_get_minimum_market_price_impact_for_auto_build add = 0.005 } }
	}
}

# ─── Auto Build Capital triggers ───
# Root is the capital location. Expects scope:building_type, scope:country.
# Reads min_discount from the per-feature priority list field
# (lsq_priority_min_discount keyed on flag:lsq_feature_auto_build_capital).
# Allows multi-level capital buildings (chancery) to be queued up to their cap:
# lsq_location_building_type_is_not_at_max_level passes when the building doesn't
# exist yet OR exists below max with no level currently under construction.
lsq_capital_location_can_auto_build = {
	scope:country = { lsq_should_pause_auto_expand = no }
	save_temporary_scope_as = location
	lsq_location_building_type_is_not_at_max_level = { target = scope:building_type }
	lsq_location_allowed_to_build = { target = scope:building_type }
	lsq_location_has_enough_pops_available = yes
	lsq_check_location_meets_min_market_access = yes
	lsq_capital_meets_min_discount = yes
	lsq_get_open_build_queue_slots > 0
	lsq_check_country_gold_above = { threshold = lsq_expected_gold_available_for_building_type_for_auto_build }
}

# Expects scopes: country, location, building_type.
# Discount setting is read from the priority list's per-feature min_discount field.
# Demand-only buildings (zero standard construction demand) always pass.
# 0.005 tolerance matches the auto_build_list discount check so a -33% slider
# accepts a -32.5% real cost adjustment without float-rounding misses.
lsq_capital_meets_min_discount = {
	OR = {
		scope:building_type = { save_temporary_scope_as = building_type lsq_get_standard_construction_demand_total = 0 }
		scope:building_type = { lsq_get_standard_construction_cost_adjustment_for_building_type <= { value = lsq_get_min_discount_for_auto_build_capital add = 0.005 } }
	}
}
