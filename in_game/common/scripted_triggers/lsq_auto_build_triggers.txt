# Generic triggers for auto-building

# Expects scopes: building_type, country, location
lsq_auto_build_check_owner_gold_available = {
	scope:country.gold > lsq_expected_gold_available_for_building_type_for_auto_build
}

# Root is the location
lsq_auto_build_check_no_building_type = {
	NOT = { has_building = $target$ }
}

}

# Root is the location. Expects the building type as target and scope country. True when location has enough RGO/lumber levels for this building type (sawmill: RGO or lumber_mill >= required; windmill/local_smelters: RGO >= required).
lsq_auto_build_check_rgo_level_requirement_met = {
	switch = { trigger = $target$
		building_type:sawmill = { lsq_auto_build_check_rgo_level_requirement_met_sawmill = yes }
		building_type:windmill = { lsq_auto_build_check_rgo_level_requirement_met_windmill = yes }
		building_type:local_smelters = { lsq_auto_build_check_rgo_level_requirement_met_local_smelters = yes }
		fallback = { always = no }
	}
}

# Expects the scope building_type. True only for sawmill, windmill, local_smelters (so tooltip row "Has required RGO levels" is shown only for those).
lsq_auto_build_show_rgo_level_tooltip_row = {
	OR = {
		scope:building_type = building_type:sawmill
		scope:building_type = building_type:windmill
		scope:building_type = building_type:local_smelters
	}
}

# Root is the country. Expects the building type as a target, also expects the scope country
lsq_auto_build_check_any_location_allowed_to_build = {
	trigger_if = { limit = { is_target_in_variable_list = { name = lsq_auto_build_building_types_in_subjects_enabled target = scope:building_type } }
		OR = {
			any_owned_location = {
				lsq_location_allowed_to_build = { target = $target$ }
			}
			any_subject_or_below = {
				any_owned_location = {
					lsq_location_allowed_to_build = { target = $target$ }
				}
			}
		}
	} trigger_else = {
		any_owned_location = {
			lsq_location_allowed_to_build = { target = $target$ }
		}
	}
}

# Root is the location. Expects the building type as a target and a trigger name, also expects the scope country
lsq_auto_build_switch_based_on_building_type = {
	switch = { trigger = $target$
		building_type:temple = { lsq_$trigger_name$_temple = yes }
		building_type:library = { lsq_$trigger_name$_library = yes }
		building_type:settlement_building = { lsq_$trigger_name$_settlement_building = yes }
		building_type:bridge_infrastructure = { lsq_$trigger_name$_bridge_infrastructure = yes }
		building_type:pound_lock_canal_infrastructure = { lsq_$trigger_name$_pound_lock_canal_infrastructure = yes }
		building_type:university = { lsq_$trigger_name$_university = yes }
		building_type:hospital = { lsq_$trigger_name$_hospital = yes }
		building_type:minting_office = { lsq_$trigger_name$_minting_office = yes }
		building_type:counting_house = { lsq_$trigger_name$_counting_house = yes }
		building_type:wharf = { lsq_$trigger_name$_wharf = yes }
		building_type:market_village = { lsq_$trigger_name$_market_village = yes }
		building_type:sawmill = { lsq_$trigger_name$_sawmill = yes }
		building_type:windmill = { lsq_$trigger_name$_windmill = yes }
		building_type:local_smelters = { lsq_$trigger_name$_local_smelters = yes }
		building_type:protected_harbor = { lsq_$trigger_name$_protected_harbor = yes }
		fallback = { always = no }
	}
}

# Root is the country. Checks if the building type's required advance has been researched.
# Used by the list visibility update to hide unresearched buildings in the settings UI.
lsq_is_building_type_advance_researched = {
	switch = { trigger = $target$
		building_type:minting_office = { has_advance = mint_advance }
		building_type:library = { has_advance = library_advance }
		building_type:pound_lock_canal_infrastructure = { has_advance = pound_lock_canals_advance }
		building_type:university = { has_advance = university_advance }
		building_type:counting_house = { has_advance = counting_house_advance }
		building_type:hospital = { has_advance = hospital_advance }
		building_type:local_smelters = { has_advance = local_smelters_advance }
		building_type:protected_harbor = { has_advance = protected_harbor_advance }
		building_type:windmill = { has_advance = windmills_advance }
		fallback = { always = yes }
	}
}
