# Root is the location, also expects the scope country
lsq_location_can_auto_build_fortress_granary = {
	AND = {
		lsq_location_will_auto_build_fortress_granary = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_fortress_granary = {
	AND = {
		scope:country = { culture = { has_culture_group = culture_group:maghrebi_group } }
		NOT = { has_building = building_type:fortress_granary }
		lsq_auto_build_check_town_or_city = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_fortress_granary = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_fortress_granary = yes
		lsq_auto_build_check_meets_minimum_control_fortress_granary = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_fortress_granary = yes
		lsq_auto_build_check_pops_available_fortress_granary = yes
		lsq_auto_build_check_can_afford_fortress_granary = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_fortress_granary = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:fortress_granary }
}

# Root is the location
lsq_auto_build_check_meets_minimum_market_price_impact_fortress_granary = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_market_price_impact_with_key = { key = fortress_granary }
}

# Root is the location
lsq_auto_build_check_pops_available_fortress_granary = {
	lsq_auto_build_check_peasants_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_fortress_granary = {
	lsq_can_afford_switch_based_on_building_type = { target = building_type:fortress_granary }
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_fortress_granary = {
	scope:country = { culture = { has_culture_group = culture_group:maghrebi_group } }
}
