# Root is the location, also expects the scope country
lsq_location_can_auto_build_counting_house = {
	AND = {
		lsq_location_will_auto_build_counting_house = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_counting_house = {
	AND = {
		lsq_auto_build_check_building_researched_counting_house = yes
		lsq_auto_build_check_no_building_type = { target = building_type:counting_house }
		lsq_auto_build_check_town_or_city = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_counting_house = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_counting_house = yes
		lsq_auto_build_check_meets_minimum_control_counting_house = yes
		lsq_auto_build_check_pops_available_counting_house = yes
		lsq_auto_build_check_can_afford_counting_house = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_counting_house = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:counting_house }
}

# Root is the location
lsq_auto_build_check_pops_available_counting_house = {
	lsq_auto_build_check_burghers_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_counting_house = {
	lsq_can_afford_counting_house = yes
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_counting_house = {
	scope:country = { has_advance = counting_house_advance }
}
