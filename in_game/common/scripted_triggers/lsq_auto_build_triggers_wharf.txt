# Root is the location, also expects the scope country
lsq_location_can_auto_build_wharf = {
	AND = {
		lsq_location_will_auto_build_wharf = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_wharf = {
	AND = {
		always = yes # not locked behind an advance
		NOT = { has_building = building_type:wharf }
		is_coastal = yes
		OR = {
			location_rank = location_rank:town
			location_rank = location_rank:city
		}
	}
	# AND = {

	# 	is_coastal = yes

	# }
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_wharf = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_wharf = yes
		lsq_auto_build_check_meets_minimum_control_wharf = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_wharf = yes
		lsq_auto_build_check_pops_available_wharf = yes
		lsq_auto_build_check_can_afford_wharf = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_wharf = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:wharf }
}

# Root is the location
lsq_auto_build_check_meets_minimum_market_price_impact_wharf = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_market_price_impact_with_key = { key = wharf }
}

# Root is the location
lsq_auto_build_check_pops_available_wharf = {
	lsq_auto_build_check_burghers_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_wharf = {
	building_type:wharf = { save_temporary_scope_as = building_type } lsq_can_afford_switch_based_on_building_type = yes
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_wharf = {
	always = yes # not locked behind an advance
}
