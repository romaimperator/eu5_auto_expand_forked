# Root is the country
lsq_should_mass_upgrade_town_to_city = {
	exists = var:lsq_mass_upgrade_town_to_city
}

# Root is the location, expects scope:country
lsq_location_can_mass_upgrade_to_city = {
	AND = {
		location_rank = location_rank:town
		population >= 30
		modifier:cannot_upgrade_location = no
		max_control >= lsq_mass_upgrade_town_to_city_min_control_scaled
		lsq_mass_upgrade_town_to_city_discount_check = yes
		scope:country = { save_temporary_scope_as = actor }
		can_build_building = building_type:lsq_upgrade_town_to_city
		scope:country = { can_build_building = building_type:lsq_upgrade_town_to_city }
	}
}

# Root is the location, expects scope:country.
# Checks that the market price impact of city upgrade construction demand is within the player's threshold.
lsq_mass_upgrade_town_to_city_discount_check = {
	save_temporary_scope_as = location
	building_type:lsq_upgrade_town_to_city = {
		lsq_get_construction_cost_adjustment_for_lsq_upgrade_town_to_city <= lsq_mass_upgrade_town_to_city_min_discount_scaled
	}
}
