# Root is the location, also expects the scope country
lsq_location_can_auto_build_confucian_academy = {
	AND = {
		lsq_location_will_auto_build_confucian_academy = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_confucian_academy = {
	AND = {
		scope:country = { has_advance = confucian_academy_advance }
		NOT = { has_building = building_type:confucian_academy }
		lsq_auto_build_check_town_or_city = yes
		lsq_auto_build_check_is_a_core = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_confucian_academy = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_confucian_academy = yes
		lsq_auto_build_check_meets_minimum_control_confucian_academy = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_confucian_academy = yes
		lsq_auto_build_check_pops_available_confucian_academy = yes
		lsq_auto_build_check_can_afford_confucian_academy = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}
