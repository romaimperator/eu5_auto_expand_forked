# Root is the location, also expects the scope country
lsq_location_can_auto_build_hospital = {
	AND = {
		lsq_location_will_auto_build_hospital = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_hospital = {
	AND = {
		lsq_auto_build_check_building_researched_hospital = yes
		lsq_auto_build_check_no_building_type = { target = building_type:hospital }
		development >= 20
		lsq_auto_build_check_town_or_city = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_hospital = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_hospital = yes
		lsq_auto_build_check_pops_available_hospital = yes
		lsq_auto_build_check_can_afford_hospital = yes
	}
}

# Root is the location
lsq_auto_build_check_pops_available_hospital = {
	lsq_auto_build_check_clergy_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_hospital = {
	lsq_can_afford_switch_based_on_building_type = { target = building_type:hospital }
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_hospital = {
	scope:country = { has_advance = hospital_advance }
}
