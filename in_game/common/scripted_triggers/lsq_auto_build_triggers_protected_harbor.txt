# Root is the location, also expects the scope country
lsq_location_can_auto_build_protected_harbor = {
	AND = {
		lsq_location_will_auto_build_protected_harbor = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location, also expects the scope country
lsq_location_allowed_to_build_protected_harbor = {
	AND = {
		scope:country = { has_advance = protected_harbor_advance }
		NOT = { has_building = building_type:protected_harbor }
		is_port = yes
		OR = {
			location_rank = location_rank:town
			location_rank = location_rank:city
		}
	}
	# AND = {
	# 	lsq_auto_build_check_building_researched_protected_harbor = yes
	# 	lsq_auto_build_check_no_building_type = { target = building_type:protected_harbor }
	# 	is_port = yes
	# 	lsq_auto_build_check_town_or_city = yes
	# }
}

# Root is the location, also expects the scope country
lsq_location_will_auto_build_protected_harbor = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_protected_harbor = yes
		lsq_auto_build_check_meets_minimum_control_protected_harbor = yes
		lsq_auto_build_check_meets_minimum_market_price_impact_protected_harbor = yes
		lsq_auto_build_check_pops_available_protected_harbor = yes
		lsq_auto_build_check_can_afford_protected_harbor = yes
		lsq_auto_build_check_location_has_more_than_zero_market_access = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_protected_harbor = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:protected_harbor }
}

# Root is the location
lsq_auto_build_check_meets_minimum_market_price_impact_protected_harbor = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_market_price_impact_with_key = { key = protected_harbor }
}

# Root is the location
lsq_auto_build_check_pops_available_protected_harbor = {
	lsq_auto_build_check_laborers_available_or_will_be = yes
}

# Root is the location
lsq_auto_build_check_can_afford_protected_harbor = {
	building_type:protected_harbor = { save_temporary_scope_as = building_type } lsq_can_afford_switch_based_on_building_type = yes
}

# Root is the location, also expects the scope country
lsq_auto_build_check_building_researched_protected_harbor = {
	scope:country = { has_advance = protected_harbor_advance }
}
