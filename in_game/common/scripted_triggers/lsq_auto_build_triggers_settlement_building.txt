# Root is the location, also expects the scope country
lsq_location_can_auto_build_settlement_building = {
	AND = {
		lsq_location_will_auto_build_settlement_building = yes
		lsq_auto_build_check_owner_gold_available = yes
	}
}

# Root is the location
lsq_location_allowed_to_build_settlement_building = {
	AND = {
		lsq_auto_build_check_building_researched_settlement_building = yes
		lsq_auto_build_check_no_building_type = { target = building_type:settlement_building }
		lsq_auto_build_check_has_less_than_5_percent_of_population_capacity = yes
		lsq_auto_build_check_rural = yes
	}
}

# Root is the location
lsq_location_will_auto_build_settlement_building = {
	AND = {
		scope:country = { lsq_should_pause_auto_expand = no }
		lsq_location_allowed_to_build_settlement_building = yes
		lsq_auto_build_check_meets_minimum_control_settlement_building = yes
		lsq_auto_build_check_pops_available_settlement_building = yes
		lsq_auto_build_check_can_afford_settlement_building = yes
	}
}

# Root is the location
lsq_auto_build_check_meets_minimum_control_settlement_building = {
	save_temporary_scope_as = location
	lsq_auto_build_check_meets_minimum_control = { target = building_type:settlement_building }
}

# Root is the location
lsq_auto_build_check_pops_available_settlement_building = {
	always = yes # settlement_buildings are so good that they should be built even if there aren't enough pops available
}

# Root is the location
lsq_auto_build_check_can_afford_settlement_building = {
	lsq_can_afford_settlement_building = yes
}

# Root is the location
lsq_auto_build_check_has_less_than_5_percent_of_population_capacity = {
	location_population_percentage < 0.05
}

# Root is the location
lsq_auto_build_check_building_researched_settlement_building = {
	always = yes # not locked behind a tech
}
