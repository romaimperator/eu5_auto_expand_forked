# Root is the country. Move scope:building_type to the top of the list (index 0). Used for shift+click on Up arrow.
lsq_auto_build_move_building_type_to_top = {
	effect = {
		set_local_variable = { name = lsq_auto_build_move_target_index value = 0 }
		lsq_auto_build_reorder_list_effect = {}
	}
}

# Root is the country. Move scope:building_type to the bottom of the list (last index). Used for shift+click on Down arrow.
lsq_auto_build_move_building_type_to_bottom = {
	effect = {
		set_local_variable = { name = types_list_size value = 0 }
		every_in_list = {
			variable = lsq_auto_build_building_types_list
			change_local_variable = { name = types_list_size add = 1 }
		}
		set_local_variable = { name = lsq_auto_build_move_target_index value = { value = local_var:types_list_size subtract = 1 } }
		lsq_auto_build_reorder_list_effect = {}
	}
}

# Root is the country. Toggles whether to auto-build the building type in scope:building_type.
lsq_toggle_auto_build_building_type = {
	effect = {
		if = { limit = { is_target_in_variable_list = { name = lsq_auto_build_building_types_enabled target = scope:building_type } }
			remove_list_variable = { name = lsq_auto_build_building_types_enabled target = scope:building_type }
		} else = {
			add_to_variable_list = { name = lsq_auto_build_building_types_enabled target = scope:building_type }
		}
	}
}

# Root is the country.
lsq_toggle_all_auto_build_building_types = {
	effect = {
		save_scope_as = country
		set_local_variable = { name = enabled_count value = 0 }
		every_in_list = {
			variable = lsq_auto_build_building_types_enabled
			change_local_variable = { name = enabled_count add = 1 }
		}

		if = { limit = { variable_list_size = { name = lsq_auto_build_building_types_list value > local_var:enabled_count } }
			every_in_list = {
				variable = lsq_auto_build_building_types_list
				save_scope_as = building_type
				scope:country = { add_to_variable_list = { name = lsq_auto_build_building_types_enabled target = scope:building_type } }
			}
		} else = {
			clear_variable_list = lsq_auto_build_building_types_enabled
		}
	}
}

# Root is the country. Expects scope building_type (sawmill/windmill/local_smelters). Increase required RGO levels before building this improvement by 1.
lsq_increase_required_rgo_levels_before_improvement_building = {
	scope = country
	effect = {
		save_scope_as = country
		if = { limit = { scope:building_type = building_type:sawmill }
			change_variable = { name = lsq_required_rgo_levels_before_sawmill add = 1 }
			clamp_variable = { name = lsq_required_rgo_levels_before_sawmill min = 0 max = 20 }
			lsq_process_building_specific_list = { building_type_text = sawmill }
		} else_if = { limit = { scope:building_type = building_type:windmill }
			change_variable = { name = lsq_required_rgo_levels_before_windmill add = 1 }
			clamp_variable = { name = lsq_required_rgo_levels_before_windmill min = 0 max = 20 }
			lsq_process_building_specific_list = { building_type_text = windmill }
		} else_if = { limit = { scope:building_type = building_type:local_smelters }
			change_variable = { name = lsq_required_rgo_levels_before_local_smelters add = 1 }
			clamp_variable = { name = lsq_required_rgo_levels_before_local_smelters min = 0 max = 20 }
			lsq_process_building_specific_list = { building_type_text = local_smelters }
		}
	}
	is_shown = {
		lsq_auto_build_show_rgo_level_tooltip_row = yes
	}
}
