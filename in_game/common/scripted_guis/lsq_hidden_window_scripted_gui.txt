lsq_hidden_window_init = {
	effect = {
		# debug_log = "LSQ: hidden window init, country is [THIS.GetCountry.GetNameWithNoTooltip]"
		lsq_auto_build_init_variables = {}

		if = { limit = { AND = { NOT = { exists = global_var:lsq_hidden_window_initialized } is_human = yes } }
			set_global_variable = { name = lsq_hidden_window_initialized value = yes }
			debug_log = "LSQ: Initializing hidden window"
			every_province = {
				if = { limit = { any_location_in_province = { count=all NOT = { exists = var:auto_expand_rgo } } }
					set_variable = { name = lsq_auto_expand_rgo_off_in_all_locations value = yes }
				} else = {
					remove_variable = lsq_auto_expand_rgo_off_in_all_locations
				}
			}
			set_variable = { name = every_building_type_count value = 0 }
			every_building_type = {
				# debug_log = "LSQ: Adding building type [THIS.GetBuildingType.GetNameWithNoTooltip] to every_building_type_list"
				root = { add_to_variable_list = { name = every_building_type_list target = prev } }
				root = { change_variable = { name = every_building_type_count add = 1 } }
			}

		# } else = {
			# debug_log = "LSQ: Hidden window already initialized or country is not human"
		}

		if = { limit = { NOT = { has_variable_list = every_building_type_list } }
			set_variable = { name = every_building_type_count value = 0 }
			every_building_type = {
				# debug_log = "LSQ: Adding building type [THIS.GetBuildingType.GetNameWithNoTooltip] to every_building_type_list"
				root = { add_to_variable_list = { name = every_building_type_list target = prev } }
				root = { change_variable = { name = every_building_type_count add = 1 } }
			}
			clear_global_variable_map = lsq_building_type_processed
			clear_global_variable_map = lsq_building_type_to_is_production_building
			clear_global_variable_map = lsq_building_type_to_is_non_production_multi_level_building
			clear_global_variable_map = lsq_building_type_to_is_one_per_location_building
			clear_global_variable_map = lsq_building_type_to_is_fort
		}

		if = { limit = { AND = { NOT = { exists = global_var:lsq_migrate_auto_expand_registered_buildings } is_human = yes } }
			debug_log = "LSQ: Migrating auto expand registered buildings"
			every_owned_location = {
				limit = { has_variable_list = auto_expand_registered_buildings }
				save_scope_as = location
				# debug_log = "LSQ: Migrating auto expand registered buildings for location [THIS.GetLocation.GetNameWithNoTooltip]"
				every_in_list = {
					variable = auto_expand_registered_buildings
					save_scope_as = building
					# debug_log = "LSQ: Migrating auto expand registered buildings for building [THIS.GetBuilding.GetNameWithNoTooltip]"
					if = { limit = { AND = { scope_type = building exists = building_type } }
						scope:location = {
							add_to_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:building.building_type }
						}
					} else_if = { limit = { scope_type = building_type }
						scope:location = {
							add_to_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:building }
						}
					}
				}
			}
			set_global_variable = { name = lsq_migrate_auto_expand_registered_buildings value = yes }
		}
	}
}

# Root is the player
lsq_should_building_type_section_show = {
	is_shown = {
		OR = {
			NOT = { has_global_variable_map = lsq_building_type_processed }
			AND = {
				has_global_variable_map = lsq_building_type_processed
				global_variable_map_size = { name = lsq_building_type_processed value < var:every_building_type_count }
			}
		}
	}
}

	is_shown = {
		AND = {
			has_global_variable_map = lsq_building_type_processed
			is_key_in_global_variable_map = { name = lsq_building_type_processed target = this }
		}
	}
