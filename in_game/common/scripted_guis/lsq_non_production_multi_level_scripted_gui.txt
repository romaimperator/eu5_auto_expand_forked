# Root is the building
lsq_set_auto_expand_for_non_production_multi_level_building = {
	effect = {
		save_scope_as = the_building
		if = { limit = { lsq_is_non_production_multi_level_building_in_auto_expand_list = yes }
			location = { remove_list_variable = { name = auto_expand_registered_non_production_multi_level_buildings target = scope:the_building } }
		} else = {
			location = { add_to_variable_list = { name = auto_expand_registered_non_production_multi_level_buildings target = scope:the_building } }
		}
	}
}

# Root is the building
lsq_is_auto_expand_active_for_non_production_multi_level_building = {
	is_shown = {
		lsq_is_non_production_multi_level_building_in_auto_expand_list = yes
	}
}

lsq_is_non_production_multi_level_building_able_to_auto_expand = {
	is_shown = {
		owner = { lsq_should_pause_auto_expand = no }
		lsq_non_production_multi_level_building_can_auto_expand_other_than_checking_for_upgrades = yes
		lsq_is_building_not_max_level_and_not_obsolete_or_is_building_type_upgradeable_and_can_be_upgraded_by_owner = yes
	}
}

lsq_check_for_auto_expand_when_upgrading_non_production_multi_level_building = {
	effect = {
		# given scope is the obsolete version of building that is being upgraded
		save_scope_as = the_building
		location = {
			if = { limit = { is_target_in_variable_list = { name = auto_expand_registered_non_production_multi_level_buildings target = scope:the_building } }
				remove_list_variable = { name = auto_expand_registered_non_production_multi_level_buildings target = scope:the_building }
				ordered_buildings_in_location = { # ordered_x always returns first scope in ordering, therefore we get the most recently built building
					order_by = building_index
					save_scope_as = the_new_building
					location = {
						if = { limit = { NOT = { is_target_in_variable_list =  { name = auto_expand_registered_non_production_multi_level_buildings target = scope:the_new_building } } }
							add_to_variable_list = { name = auto_expand_registered_non_production_multi_level_buildings target = scope:the_new_building }
						}
					}
				}
			}
		}
	}
}
