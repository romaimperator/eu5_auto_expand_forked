# Food auto-build CMM list change handler.
# Fires when any field in the auto_food_list changes (enable/disable, reorder, river_only toggle).
lsq__auto_food_list_on_changed = {
	scope = country
	is_shown = { always = yes }
	effect = {
		cmm_apply_list_change = { setting = lsq__auto_food_list }
		lsq_rebuild_auto_food_lookups = yes
	}
}

# Root is location; returns true if location is a rural settlement with one of the 14 food good RGOs.
# Used by the location window widget to gate visibility of the auto-food button — towns and cities
# can't build the food-specific buildings (farming village, fishing village, etc.) so the button is
# hidden there.
lsq_is_food_rgo_location = {
	is_shown = {
		location_rank = location_rank:rural_settlement
		OR = {
			raw_material = goods:rice
			raw_material = goods:wheat
			raw_material = goods:maize
			raw_material = goods:potato
			raw_material = goods:livestock
			raw_material = goods:millet
			raw_material = goods:legumes
			raw_material = goods:wool
			raw_material = goods:fish
			raw_material = goods:olives
			raw_material = goods:fruit
			raw_material = goods:wild_game
			raw_material = goods:beeswax
			raw_material = goods:fur
		}
	}
}

# Root is location. Expects scope:food_good, scope:country (owner). Optional scope:set_off (bool).
# Toggles per-location auto_food state, mirroring lsq_set_auto_expand_for_rgo's dual-flag logic:
#   - If the location is currently "covered by mass" (good enabled AND river filter passes),
#     flip auto_food_location_excluded (opt-out).
#   - Otherwise flip auto_food_location_enabled (opt-in).
lsq_set_auto_food_for_location = {
	effect = {
		if = {
			limit = {
				scope:country = { is_target_in_variable_list = { name = lsq_auto_food_rgo_types_enabled target = scope:food_good } }
				lsq_food_location_meets_river_filter = yes
			}
			# Covered by mass: toggle the excluded flag
			if = { limit = { NOT = { exists = scope:set_off } }
				if = { limit = { exists = var:auto_food_location_excluded }
					remove_variable = auto_food_location_excluded
				} else = {
					set_variable = { name = auto_food_location_excluded value = yes }
				}
			} else = {
				if = { limit = { scope:set_off = yes }
					if = { limit = { NOT = { exists = var:auto_food_location_excluded } }
						set_variable = { name = auto_food_location_excluded value = yes }
					}
				} else = {
					remove_variable = auto_food_location_excluded
				}
			}
		} else = {
			# Not covered by mass: toggle the enabled flag.
			# The enabled flag marks a manual per-location opt-in, so it also locks
			# the location against rank upgrades via lsq_auto_food_locked_location.
			if = { limit = { NOT = { exists = scope:set_off } }
				if = { limit = { exists = var:auto_food_location_enabled }
					remove_variable = auto_food_location_enabled
					remove_location_modifier = lsq_auto_food_locked_location
				} else = {
					set_variable = { name = auto_food_location_enabled value = yes }
					add_location_modifier = { modifier = lsq_auto_food_locked_location years = 1000 mode = replace }
				}
			} else = {
				if = { limit = { scope:set_off = yes }
					remove_variable = auto_food_location_enabled
					remove_location_modifier = lsq_auto_food_locked_location
				} else = {
					set_variable = { name = auto_food_location_enabled value = yes }
					add_location_modifier = { modifier = lsq_auto_food_locked_location years = 1000 mode = replace }
				}
			}
		}
	}

	is_shown = {
		OR = {
			exists = var:auto_food_location_enabled
			AND = {
				scope:country = { is_target_in_variable_list = { name = lsq_auto_food_rgo_types_enabled target = scope:food_good } }
				lsq_food_location_meets_river_filter = yes
				NOT = { exists = var:auto_food_location_excluded }
			}
		}
	}
}

# ─── Glow / tooltip gates ───

# Root is country.
# True if auto-food is active in the country's priority feature list.
lsq_is_auto_food_feature_enabled = {
	is_shown = {
		is_target_in_variable_list = { name = lsq_priority_features_enabled target = flag:lsq_feature_auto_food }
	}
}

# Root is location.
# Thin wrapper so the tooltip row can call a scripted_gui.
lsq_food_location_has_market_access = {
	is_shown = {
		lsq_check_location_meets_min_market_access = yes
	}
}

# Root is location. Expects scope:country.
# True when neither laborers nor peasants are scarce — i.e., the location isn't still
# promoting pops to fill existing jobs. Treated as a single combined check because the
# build logic skips the location whenever either pool is scarce (peasants promote to
# laborers, so a peasant-consuming build would starve the RGO).
lsq_food_location_pops_not_still_filling = {
	is_shown = {
		NOT = { lsq_food_location_pops_still_filling = yes }
	}
}

# Root is location. Expects scope:country, scope:food_good.
# True if at least one of the seven food-building options is viable right now
# (can-build AND affordable AND meets minimum discount).
lsq_food_location_has_viable_option = {
	is_shown = {
		lsq_food_location_has_viable_option = yes
	}
}

# Root is location. Expects scope:country, scope:food_good.
# True if at least one food-building option could physically be built here
# (can-build only; ignores gold/discount). Used to disambiguate the "no option"
# tooltip row from the "no option meets min discount" tooltip row.
lsq_food_location_has_buildable_option = {
	is_shown = {
		lsq_food_location_has_buildable_option = yes
	}
}

# Root is location. Expects scope:country, scope:food_good.
# Top-level glow gate for the per-location auto-food button. Mirrors the preconditions
# inside lsq_food_auto_build_in_location, minus transient/queue conditions.
lsq_is_able_to_auto_food_location = {
	is_shown = {
		scope:country = {
			lsq_should_pause_auto_expand = no
			is_target_in_variable_list = { name = lsq_priority_features_enabled target = flag:lsq_feature_auto_food }
		}
		lsq_check_location_meets_min_market_access = yes
		NOT = { lsq_food_location_pops_still_filling = yes }
		lsq_food_location_has_viable_option = yes
	}
}
