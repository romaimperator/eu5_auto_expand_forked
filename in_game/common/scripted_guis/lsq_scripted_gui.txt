lsq_set_auto_expand_for_building = {
	effect = {
		save_scope_as = the_building
		location = {
			if = { limit = { is_target_in_variable_list = { name = auto_expand_registered_buildings target = scope:the_building } }
				remove_list_variable = { name = auto_expand_registered_buildings target = scope:the_building }
			} else = {
				add_to_variable_list = { name = auto_expand_registered_buildings target = scope:the_building }
			}
		}
	}
}

lsq_check_for_auto_expand_when_upgrading_building = {
	effect = {
		# given scope is the obsolete version of building that is being upgraded
		save_scope_as = the_building
		location = {
			if = { limit = { is_target_in_variable_list = { name = auto_expand_registered_buildings target = scope:the_building } }
				remove_list_variable = { name = auto_expand_registered_buildings target = scope:the_building }
				ordered_buildings_in_location = { # ordered_x always returns first scope in ordering, therefore we get the most recently built building
					order_by = building_index
					save_scope_as = the_new_building
					location = {
						if = { limit = { NOT = { is_target_in_variable_list =  { name = auto_expand_registered_buildings target = scope:the_new_building } } }
							add_to_variable_list = { name = auto_expand_registered_buildings target = scope:the_new_building }
						}
					}
				}
			}
		}
	}
}
lsq_check_for_auto_expand_when_removing_building = {
	effect = {
		save_scope_as = the_building
		location = {
			if = { limit = { AND = { scope:the_building = { building_level < 2 } is_target_in_variable_list = { name = auto_expand_registered_buildings target = scope:the_building } } }
				remove_list_variable = { name = auto_expand_registered_buildings target = scope:the_building }
			}
		}
	}
}
lsq_is_auto_expand_visible_for_building = {
	is_shown = {
		building_type = {
			OR = {
				lsq_is_building_type_production_building = yes
				lsq_is_building_type_non_production_multi_level_building = yes
			}
		}
	}
}

lsq_is_auto_expand_active_for_building = {
	is_shown = {
		save_temporary_scope_as = the_building
		location = { is_target_in_variable_list = { name = auto_expand_registered_buildings target = scope:the_building } }
	}
}

lsq_show_settings_window = {
	scope = country
	effect = {
		# clear_global_variable_map = lsq_building_type_to_is_production_building
		# clear_global_variable_map = lsq_building_type_to_is_non_production_multi_level_building
		# clear_global_variable_map = lsq_building_type_to_is_one_per_location_building
		set_variable = { name = lsq_settings_window_visible value = yes }
		if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_buildings } }
			set_variable = { name = lsq_minimum_gold_for_buildings value = 0 }
		}
		if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_rgos } }
			set_variable = { name = lsq_minimum_gold_for_rgos value = 0 }
		}
		# every_building_type = {
		# 	root = { add_to_variable_list = { name = every_building_type_list target = prev } }
		# }

		add_to_variable_list = { name = lsq_auto_build_building_types_list target = building_type:temple }

		clear_variable_list = lsq_all_controlled_locations_list
		ordered_controlled_location = {
			order_by = max_control
			save_scope_as = location
			max = 1000000
			check_range_bounds = no
			root = { add_to_variable_list = { name = lsq_all_controlled_locations_list target = scope:location } }
		}
	}

	is_shown = {
		exists = var:lsq_settings_window_visible
	}
}

lsq_should_auto_expand_upgrade_buildings = {
	effect = {
		if = { limit = { exists = var:lsq_should_not_upgrade_buildings }
			remove_variable = lsq_should_not_upgrade_buildings
		} else = {
			set_variable = { name = lsq_should_not_upgrade_buildings value = yes }
		}
	}

	is_shown = {
		lsq_should_auto_expand_upgrade_buildings = yes
	}
}
