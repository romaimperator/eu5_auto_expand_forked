lsq_set_auto_expand_for_building = {
	effect = {
		save_scope_as = the_building
		building_type = { save_scope_as = building_type }
		location = {
			if = { limit = { NOT = { exists = scope:set_off } }
				if = { limit = { is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:building_type } }
					remove_list_variable = { name = lsq_auto_expand_registered_building_types target = scope:building_type }
				} else = {
					add_to_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:building_type }
				}
			} else = {
				if = { limit = { scope:set_off = yes }
					# debug_log = "LSQ: scope:set_off is yes"
					remove_list_variable = { name = lsq_auto_expand_registered_building_types target = scope:building_type }
				} else = {
					# debug_log = "LSQ: scope:set_off is no"
					add_to_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:building_type }
				}
			}
		}
	}
}

lsq_check_for_auto_expand_when_upgrading_building = {
	effect = {
		# given scope is the obsolete version of building that is being upgraded
		save_scope_as = the_building
		scope:the_building = { building_type = { save_temporary_scope_as = old_building_type } }
		location = {
			if = { limit = { is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:old_building_type } }
				remove_list_variable = { name = lsq_auto_expand_registered_building_types target = scope:old_building_type }
				ordered_buildings_in_location = { # ordered_x always returns first scope in ordering, therefore we get the most recently built building
					order_by = building_index
					save_scope_as = the_new_building
					scope:the_new_building = {
						building_type = { save_temporary_scope_as = new_building_type }
					}
					location = {
						if = { limit = { NOT = { is_target_in_variable_list =  { name = lsq_auto_expand_registered_building_types target = scope:new_building_type } } }
							add_to_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:new_building_type }
						}
					}
				}
			}
		}
	}
}
lsq_check_for_auto_expand_when_removing_building = {
	effect = {
		save_scope_as = the_building
		scope:the_building = { building_type = { save_temporary_scope_as = building_type } }
		location = {
			if = { limit = { AND = { scope:the_building = { building_level < 2 } is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:building_type } } }
				remove_list_variable = { name = lsq_auto_expand_registered_building_types target = scope:building_type }
			}
		}
	}
}
