lsq_set_auto_expand_for_building = {
	effect = {
		save_scope_as = the_building
		building_type = { save_scope_as = building_type }
		location = {
			if = { limit = { owner = { is_target_in_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:building_type } } }
				# Mass mode: operate on exclusion list (inverted logic)
				if = { limit = { NOT = { exists = scope:set_off } }
					if = { limit = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:building_type } }
						remove_list_variable = { name = lsq_auto_expand_excluded_building_types target = scope:building_type }
					} else = {
						add_to_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:building_type }
					}
				} else = {
					if = { limit = { scope:set_off = yes }
						if = { limit = { NOT = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:building_type } } }
							add_to_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:building_type }
						}
					} else = {
						remove_list_variable = { name = lsq_auto_expand_excluded_building_types target = scope:building_type }
					}
				}
			} else = {
				# Normal mode: operate on registration list
				if = { limit = { NOT = { exists = scope:set_off } }
					if = { limit = { is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:building_type } }
						remove_list_variable = { name = lsq_auto_expand_registered_building_types target = scope:building_type }
					} else = {
						add_to_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:building_type }
					}
				} else = {
					if = { limit = { scope:set_off = yes }
						remove_list_variable = { name = lsq_auto_expand_registered_building_types target = scope:building_type }
					} else = {
						add_to_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:building_type }
					}
				}
			}
		}
	}
}

lsq_check_for_auto_expand_when_upgrading_building = {
	effect = {
		# given scope is the obsolete version of building that is being upgraded
		save_scope_as = the_building
		scope:the_building = { building_type = { save_temporary_scope_as = old_building_type } }
		location = {
			if = { limit = { owner = { is_target_in_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:old_building_type } } }
				# Mass mode: add new type to country mass list, swap exclusion entries
				ordered_buildings_in_location = {
					order_by = building_index
					save_scope_as = the_new_building
					scope:the_new_building = {
						building_type = { save_temporary_scope_as = new_building_type }
					}
					owner = {
						if = { limit = { NOT = { is_target_in_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:new_building_type } } }
							add_to_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:new_building_type }
						}
					}
					location = {
						if = { limit = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:old_building_type } }
							remove_list_variable = { name = lsq_auto_expand_excluded_building_types target = scope:old_building_type }
							if = { limit = { NOT = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:new_building_type } } }
								add_to_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:new_building_type }
							}
						}
					}
				}
			} else = {
				# Normal mode: swap registration from old to new
				if = { limit = { is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:old_building_type } }
					remove_list_variable = { name = lsq_auto_expand_registered_building_types target = scope:old_building_type }
					ordered_buildings_in_location = {
						order_by = building_index
						save_scope_as = the_new_building
						scope:the_new_building = {
							building_type = { save_temporary_scope_as = new_building_type }
						}
						location = {
							if = { limit = { NOT = { is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:new_building_type } } }
								add_to_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:new_building_type }
							}
						}
					}
				}
			}
		}
	}
}
# Called from the hidden window GUI to perform a building upgrade.
# Root = location, scope:old_building_type = obsolete type, scope:new_building_type = upgrade target (from GetNextReplaced), scope:country = player
lsq_gui_construct_upgrade = {
	is_valid = {
		scope:country = { has_variable = lsq_gui_upgrades_pending }
		is_target_in_variable_list = { name = lsq_gui_upgrade_building_types target = scope:old_building_type }
		OR = {
			scope:country = { lsq_should_auto_expand_upgrade_buildings = yes }
			scope:country = { lsq_should_upgrade_all_buildings = yes }
		}
		lsq_auto_expand_check_minimum_gold_reserve = yes
	}
	effect = {
		construct_building = { building_type = scope:new_building_type }
		# Swap auto-expand tracking from old to new
		if = { limit = { owner = { is_target_in_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:old_building_type } } }
			# Mass mode: add new type to country mass list and swap exclusion entries
			owner = {
				if = { limit = { NOT = { is_target_in_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:new_building_type } } }
					add_to_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:new_building_type }
				}
			}
			if = { limit = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:old_building_type } }
				remove_list_variable = { name = lsq_auto_expand_excluded_building_types target = scope:old_building_type }
				if = { limit = { NOT = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:new_building_type } } }
					add_to_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:new_building_type }
				}
			}
		} else = {
			# Normal mode: swap registration entries (only if old type was registered)
			if = { limit = { is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:old_building_type } }
				remove_list_variable = { name = lsq_auto_expand_registered_building_types target = scope:old_building_type }
				if = { limit = { NOT = { is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:new_building_type } } }
					add_to_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:new_building_type }
				}
			}
		}
		# Track that this building type has an upgrade queued so the monthly script won't re-flag it
		if = { limit = { NOT = { is_target_in_variable_list = { name = lsq_upgrade_in_progress target = scope:old_building_type } } }
			add_to_variable_list = { name = lsq_upgrade_in_progress target = scope:old_building_type }
		}
		# Clear from pending list
		remove_list_variable = { name = lsq_gui_upgrade_building_types target = scope:old_building_type }
	}
}

}

# Root is the building type. Toggles mass auto-expand for this specific building type.
# When ON: this building type auto-expands at all owned locations (exclusion-based per location).
# When OFF: this building type is removed from mass tracking and location exclusion lists are cleaned.
lsq_toggle_mass_auto_expand_for_building_type = {
	saved_scopes = { country }
	effect = {
		save_scope_as = building_type
		if = { limit = { scope:country = { is_target_in_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:building_type } } }
			# Turning OFF — remove from country list and clear exclusion entries for this type
			scope:country = {
				remove_list_variable = { name = lsq_mass_auto_expand_building_types target = scope:building_type }
				ordered_owned_location = {
					order_by = max_control
					max = lsq_max_guess_at_number_of_locations
					check_range_bounds = no
					limit = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:building_type } }
					remove_list_variable = { name = lsq_auto_expand_excluded_building_types target = scope:building_type }
				}
			}
		} else = {
			# Turning ON — add to country list and clear per-location registrations for this type
			scope:country = {
				add_to_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:building_type }
				ordered_owned_location = {
					order_by = max_control
					max = lsq_max_guess_at_number_of_locations
					check_range_bounds = no
					limit = { is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = scope:building_type } }
					remove_list_variable = { name = lsq_auto_expand_registered_building_types target = scope:building_type }
				}
			}
		}
	}
	is_shown = {
		save_temporary_scope_as = building_type
		scope:country = { is_target_in_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:building_type } }
	}
}
