lsq_set_auto_expand_for_rgo = {
	effect = {
		# market = { add_goods_supply = { goods = goods:lumber amount = -1 } }
		if = { limit = { exists = var:auto_expand_rgo }
			remove_variable = auto_expand_rgo
			# If after turning off auto expand RGO in a location, there are no other locations within the same province, set the province-level variable
			province = {
				if = { limit = { any_location_in_province = { count=all NOT = { exists = var:auto_expand_rgo } } }
					set_variable = { name = lsq_auto_expand_rgo_off_in_all_locations value = yes }
				}
			}
		} else = {
			set_variable = { name = auto_expand_rgo value = yes }
			# Remove the province-level variable to indicate that there is at least one location with auto expand RGO on
			province = { remove_variable = lsq_auto_expand_rgo_off_in_all_locations }
		}
		# owner = { remove_variable = lsq_update_rgo_information }
	}

	is_shown = {
		exists = var:auto_expand_rgo
	}
}

lsq_is_able_to_auto_expand_rgo = {
	is_shown = {
		lsq_is_able_to_auto_expand_rgo = yes
	}
}

		lsq_is_rgo_not_max_level_already = yes
	}

lsq_is_rgo_fully_employed = {
	is_shown = {
		# Root scope is a location (passed from GUI).
		# True if the location's RGO building(s) are at full capacity (or none exists yet at level 0),
		# OR if the location is not in a negative labourer deficit (i.e. can be filled, including imminent promotion).
		lsq_is_rgo_fully_employed = yes
	}
}
