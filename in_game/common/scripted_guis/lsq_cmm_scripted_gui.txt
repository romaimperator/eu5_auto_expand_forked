# Settings (settings) Tab
## Priority (priority) Group
lsq__priority_list_on_changed = {
	scope = country

	is_shown = {
		always = yes
	}

	effect = {
		cmm_apply_list_change = {
			setting = lsq__priority_list
		}
		lsq_rebuild_priority_lookups = yes
	}
}
