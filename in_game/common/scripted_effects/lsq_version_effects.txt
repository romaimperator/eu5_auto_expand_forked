# Saves the version of the mod that was used to start the game to the global variables
lsq_set_mod_version_to_current = {
  lsq_set_mod_version_to = { major = lsq_major_version minor = lsq_minor_version patch = lsq_patch_version }
}

lsq_set_mod_version_to = {
  set_global_variable = { name = lsq_major_version value = $major$ }
  set_global_variable = { name = lsq_minor_version value = $minor$ }
  set_global_variable = { name = lsq_patch_version value = $patch$ }
}
