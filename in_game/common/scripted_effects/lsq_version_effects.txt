# Saves the version of the mod that was used to start the game
# Expects scopes: country
lsq_set_mod_version_to_current = {
  lsq_set_mod_version_to = { major = lsq_major_version minor = lsq_minor_version patch = lsq_patch_version }
}

# Expects scopes: country
lsq_set_mod_version_to = {
  scope:country = {
    set_variable = { name = lsq_major_version value = $major$ }
    set_variable = { name = lsq_minor_version value = $minor$ }
    set_variable = { name = lsq_patch_version value = $patch$ }
  }
}

# Expects scopes: old_country, new_country
lsq_copy_mod_version_from_old_country_to_new_country = {
  scope:new_country = {
    set_variable = { name = lsq_major_version value = scope:old_country.var:lsq_major_version }
    set_variable = { name = lsq_minor_version value = scope:old_country.var:lsq_minor_version }
    set_variable = { name = lsq_patch_version value = scope:old_country.var:lsq_patch_version }
  }
}
