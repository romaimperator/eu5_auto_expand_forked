# Sync all auto-build list values from CMM to legacy variables.
# Scope: country
lsq_sync_auto_build_list_to_legacy = {
	clear_variable_list = lsq_auto_build_building_types_list
	clear_variable_list = lsq_auto_build_building_types_enabled
	clear_variable_list = lsq_auto_build_building_types_in_subjects_enabled
	cmm_for_each_list_item = {
		setting = lsq__auto_build_list
		effect = lsq_sync_auto_build_list_item
	}
}

# Called by cmm_for_each_list_item to sync one item to legacy variables.
# Iterates in display order so lsq_auto_build_building_types_list reflects the CMM ordering.
# Scope: country
# Args: setting (key), i (number)
# scope:cmm_list_current_item_value: building_type scope for this item
lsq_sync_auto_build_list_item = {
	# Add to ordered list (iteration order = display order)
	add_to_variable_list = { name = lsq_auto_build_building_types_list target = scope:cmm_list_current_item_value }

	# Build flag keys for field access (can't use $i$ inside "..." quoted expressions)
	set_local_variable = { name = lsq_key_f1 value = flag:$setting$_i$i$_f1 }
	set_local_variable = { name = lsq_key_f2 value = flag:$setting$_i$i$_f2 }
	set_local_variable = { name = lsq_key_f3 value = flag:$setting$_i$i$_f3 }
	set_local_variable = { name = lsq_key_f4 value = flag:$setting$_i$i$_f4 }
	set_local_variable = { name = lsq_key_f5 value = flag:$setting$_i$i$_f5 }

	# Sync enabled
	if = {
		limit = { "variable_map(cmm|local_var:lsq_key_f1)" >= 1 }
		add_to_variable_list = { name = lsq_auto_build_building_types_enabled target = scope:cmm_list_current_item_value }
	}

	# Sync in_subjects
	if = {
		limit = { "variable_map(cmm|local_var:lsq_key_f2)" >= 1 }
		add_to_variable_list = { name = lsq_auto_build_building_types_in_subjects_enabled target = scope:cmm_list_current_item_value }
	}

	# Sync min_control (CMM: 0-100 → legacy: 0-1)
	set_variable = { name = lsq_tmp value = "variable_map(cmm|local_var:lsq_key_f3)" }
	change_variable = { name = lsq_tmp multiply = 0.01 }
	add_to_variable_map = { name = lsq_minimum_control_for_auto_build key = scope:cmm_list_current_item_value value = var:lsq_tmp }

	# Sync min_discount (CMM: -33..33 → legacy: -0.33..0.33)
	set_variable = { name = lsq_tmp value = "variable_map(cmm|local_var:lsq_key_f4)" }
	change_variable = { name = lsq_tmp multiply = 0.01 }
	add_to_variable_map = { name = lsq_minimum_market_price_impact_for_auto_build key = scope:cmm_list_current_item_value value = var:lsq_tmp }

	# Sync min_rgos (only for improvement buildings)
	if = {
		limit = { scope:cmm_list_current_item_value = building_type:sawmill }
		set_variable = { name = lsq_required_rgo_levels_before_sawmill value = "variable_map(cmm|local_var:lsq_key_f5)" }
	}
	if = {
		limit = { scope:cmm_list_current_item_value = building_type:windmill }
		set_variable = { name = lsq_required_rgo_levels_before_windmill value = "variable_map(cmm|local_var:lsq_key_f5)" }
	}
	if = {
		limit = { scope:cmm_list_current_item_value = building_type:local_smelters }
		set_variable = { name = lsq_required_rgo_levels_before_local_smelters value = "variable_map(cmm|local_var:lsq_key_f5)" }
	}

	# Benign warning suppression
	if = {
		limit = { always = no }
		cmf_suppress = { v = lsq_key_f1 }
		cmf_suppress = { v = lsq_key_f2 }
		cmf_suppress = { v = lsq_key_f3 }
		cmf_suppress = { v = lsq_key_f4 }
		cmf_suppress = { v = lsq_key_f5 }
	}
}

# Callback handler for cmf_on_callback.
# var:cmf_callback is the flag of the setting, alert, or action bar element that was interacted with.
# Scope: country
lsq_handle_cmf_callback = {
	switch = {
		trigger = var:cmf_callback
		# Construction (construction) Tab
		## Settings (settings) Group
		flag:lsq__pause_auto_expand = {
			cmm_sync_bool_alias = {
				setting = lsq__pause_auto_expand
				alias = lsq_pause_auto_expand
			}
		}
		flag:lsq__should_not_upgrade_buildings = {
			cmm_sync_bool_alias_inverted = {
				setting = lsq__should_not_upgrade_buildings
				alias = lsq_should_not_upgrade_buildings
			}
		}
		flag:lsq__minimum_gold_for_buildings = {
			cmm_sync_setting_alias = {
				setting = lsq__minimum_gold_for_buildings
				alias = lsq_minimum_gold_for_buildings
			}
		}
		flag:lsq__minimum_gold_for_rgos = {
			cmm_sync_setting_alias = {
				setting = lsq__minimum_gold_for_rgos
				alias = lsq_minimum_gold_for_rgos
			}
		}
		flag:lsq__minimum_gold_for_auto_build = {
			cmm_sync_setting_alias = {
				setting = lsq__minimum_gold_for_auto_build
				alias = lsq_minimum_gold_for_auto_build
			}
		}
		flag:lsq__allowed_building_overbuild_levels = {
			cmm_sync_setting_alias = {
				setting = lsq__allowed_building_overbuild_levels
				alias = lsq_allowed_building_overbuild_levels
			}
		}
		## Auto-Build (auto_build) Group
		flag:lsq__auto_build_list = {
			lsq_sync_auto_build_list_to_legacy = yes
		}
	}
}

# Benign warning suppression for variables/flags set or used at runtime
# via GUI scripted effects, AddScope, variable maps, or on_actions.
_lsq_suppress_warnings = {
	if = {
		limit = { always = no }
		# Variables used but set at runtime via GUI scripted effects or on_actions
		cmf_suppress = { v = lsq_building_type_processed }
		cmf_suppress = { v = auto_expand_registered_buildings }
		cmf_suppress = { v = lsq_run_auto_expand_rgo }
		cmf_suppress = { v = every_building_type_list }
		cmf_suppress = { v = every_building_type_count }
		cmf_suppress = { v = lsq_hidden_window_initialized }
		cmf_suppress = { v = lsq_auto_expand_rgo_off_in_all_locations }
		# Variable set but used via GUI datamodel
		cmf_suppress = { v = every_non_standard_building_type }
		# Event targets set via GUI AddScope
		cmf_suppress = { v = skip }
		cmf_suppress = { v = set_off }
	}
}
