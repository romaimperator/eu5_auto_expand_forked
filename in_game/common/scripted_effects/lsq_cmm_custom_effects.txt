# Custom CMM integration code not managed by the visual editor.
# Called after lsq_register_cmf_mod (visual-editor-managed) completes.

# Post-registration: override default values and build lookup tables.
# Scope: country
lsq_cmm_post_registration = {
	# Override min_control defaults: temple(1), minting_office(2), settlement_building(3) should start at 0
	# (they increase control / population growth, so best to build everywhere regardless of control)
	if = {
		limit = { NOT = { has_variable = lsq_cmm_auto_build_list_initialized } }
		set_variable = { name = lsq_cmm_auto_build_list_initialized value = 1 }
		add_to_variable_map = { name = cmm key = flag:lsq__auto_build_list_i1_f3 value = 0 }
		add_to_variable_map = { name = cmm key = flag:lsq__auto_build_list_i2_f3 value = 0 }
		add_to_variable_map = { name = cmm key = flag:lsq__auto_build_list_i3_f3 value = 0 }
	}

	# Override priority list defaults: enable buildings (item 1) and RGOs (item 2) by default
	if = {
		limit = { NOT = { has_variable = lsq_cmm_priority_list_initialized } }
		set_variable = { name = lsq_cmm_priority_list_initialized value = 1 }
		add_to_variable_map = { name = cmm key = flag:lsq__priority_list_i1_f3 value = 1 }
		add_to_variable_map = { name = cmm key = flag:lsq__priority_list_i2_f3 value = 1 }
	}

	# Build lookup tables from CMM list fields
	lsq_rebuild_priority_lookups = yes
	lsq_rebuild_auto_build_lookups = yes
	lsq_rebuild_auto_food_lookups = yes

	# Hide buildings whose advance hasn't been researched yet
	lsq_update_auto_build_list_visibility = yes
}

# Rebuild priority feature lookup tables from CMM list field data.
# Called on registration and on every setting change.
# Scope: country
lsq_rebuild_priority_lookups = {
	if = {
		limit = { NOT = { has_variable_list = cmm_list_items_lsq__priority_list } }
	}
	else = {

	# Ordered features list (for monthly pulse iteration in priority order)
	cmm_build_list_ordered_values = { setting = lsq__priority_list list_name = lsq_priority_features_list }

	# Min profit by feature (numeric field slot 1, direct gold values)
	cmm_build_list_field_map = { setting = lsq__priority_list field_slot = 1 map_name = lsq_priority_min_profit multiply = 1 }

	# Min discount by feature (numeric field slot 2, CMM -33..33 → -0.33..0.33)
	cmm_build_list_field_map = { setting = lsq__priority_list field_slot = 2 map_name = lsq_priority_min_discount multiply = 0.01 }

	# Enabled features (bool field slot 3)
	cmm_build_list_bool_list = { setting = lsq__priority_list field_slot = 3 list_name = lsq_priority_features_enabled }

	} # end guard
}

# Rebuild all lookup tables from CMM list field data.
# Called on registration and on every setting change.
# Scope: country
lsq_rebuild_auto_build_lookups = {
	# Guard: skip if CMM list data doesn't exist yet (first registration may not have completed)
	if = {
		limit = { NOT = { has_variable_list = cmm_list_items_lsq__auto_build_list } }
		# List not initialized yet — lookups will be built on next registration cycle
	}
	else = {

	# Ordered building types list (for monthly pulse iteration in priority order)
	cmm_build_list_ordered_values = { setting = lsq__auto_build_list list_name = lsq_auto_build_building_types_list }

	# Enabled building types (bool field slot 1)
	cmm_build_list_bool_list = { setting = lsq__auto_build_list field_slot = 1 list_name = lsq_auto_build_building_types_enabled }

	# Subjects-enabled building types (bool field slot 2)
	cmm_build_list_bool_list = { setting = lsq__auto_build_list field_slot = 2 list_name = lsq_auto_build_building_types_in_subjects_enabled }

	# Min control by building type (numeric field slot 3, CMM 0-100 → 0-1)
	cmm_build_list_field_map = { setting = lsq__auto_build_list field_slot = 3 map_name = lsq_min_control_by_building_type multiply = 0.01 }

	# Min discount by building type (numeric field slot 4, CMM -33..33 → -0.33..0.33)
	cmm_build_list_field_map = { setting = lsq__auto_build_list field_slot = 4 map_name = lsq_min_discount_by_building_type multiply = 0.01 }

	# Min RGOs by building type (numeric field slot 5, direct values)
	cmm_build_list_field_map = { setting = lsq__auto_build_list field_slot = 5 map_name = lsq_min_rgos_by_building_type multiply = 1 }

	} # end guard
}

# Rebuild all food auto-build lookup tables from CMM list field data.
# Called on registration and on every setting change.
# Scope: country
lsq_rebuild_auto_food_lookups = {
	if = {
		limit = { NOT = { has_variable_list = cmm_list_items_lsq__auto_food_list } }
	}
	else = {

	# Ordered food RGO types list (for daily tick iteration in priority order)
	cmm_build_list_ordered_values = { setting = lsq__auto_food_list list_name = lsq_auto_food_rgo_types_list }

	# Enabled food RGO types (bool field slot 2, after data column)
	cmm_build_list_bool_list = { setting = lsq__auto_food_list field_slot = 2 list_name = lsq_auto_food_rgo_types_enabled }

	# River-only food RGO types (bool field slot 3)
	cmm_build_list_bool_list = { setting = lsq__auto_food_list field_slot = 3 list_name = lsq_auto_food_rgo_types_river_only }

	# Subjects-enabled food RGO types (bool field slot 4)
	cmm_build_list_bool_list = { setting = lsq__auto_food_list field_slot = 4 list_name = lsq_auto_food_rgo_types_in_subjects_enabled }

	} # end guard
}

# Hide/show auto-build list rows based on whether the building's advance is researched.
# Scope: country
lsq_update_auto_build_list_visibility = {
	if = {
		limit = { has_variable_list = cmm_list_items_lsq__auto_build_list }
		cmm_for_each_list_item = {
			setting = lsq__auto_build_list
			effect = lsq_auto_build_list_update_visibility_item
		}
	}
}

# Callback for cmm_for_each_list_item. Shows or hides item $i$ based on advance status.
# $i$ = item position in the list. $setting$ = composite key. scope:cmm_list_current_item_value = building_type.
# Root: country
lsq_auto_build_list_update_visibility_item = {
	if = {
		limit = { lsq_is_building_type_advance_researched = { target = scope:cmm_list_current_item_value } }
		cmm_show_list_item = { mod_id = lsq setting_id = auto_build_list item = $i$ }
	}
	else = {
		cmm_hide_list_item = { mod_id = lsq setting_id = auto_build_list item = $i$ }
	}

	# Accept $setting$ param from cmm_for_each_list_item (unused here)
	if = {
		limit = { always = no }
		cmf_suppress = { v = $setting$_$i$ }
	}
}
