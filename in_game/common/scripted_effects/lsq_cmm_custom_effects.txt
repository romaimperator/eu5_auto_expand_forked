# Custom CMM integration code not managed by the visual editor.
# Called after lsq_register_cmf_mod (visual-editor-managed) completes.

# Post-registration: override default values and build lookup tables.
# Scope: country
lsq_cmm_post_registration = {
	# Override min_control defaults: temple(1), minting_office(2), settlement_building(3) should start at 0
	# (they increase control / population growth, so best to build everywhere regardless of control)
	if = {
		limit = { NOT = { has_variable = lsq_cmm_auto_build_list_initialized } }
		set_variable = { name = lsq_cmm_auto_build_list_initialized value = 1 }
		add_to_variable_map = { name = cmm key = flag:lsq__auto_build_list_i1_f3 value = 0 }
		add_to_variable_map = { name = cmm key = flag:lsq__auto_build_list_i2_f3 value = 0 }
		add_to_variable_map = { name = cmm key = flag:lsq__auto_build_list_i3_f3 value = 0 }
	}

	# Build lookup tables from CMM list fields
	lsq_rebuild_auto_build_lookups = yes
}

# Rebuild all lookup tables from CMM list field data.
# Called on registration and on every setting change.
# Scope: country
lsq_rebuild_auto_build_lookups = {
	# Ordered building types list (for monthly pulse iteration in priority order)
	cmm_build_list_ordered_values = { setting = lsq__auto_build_list list_name = lsq_auto_build_building_types_list }

	# Enabled building types (bool field slot 1)
	cmm_build_list_bool_list = { setting = lsq__auto_build_list field_slot = 1 list_name = lsq_auto_build_building_types_enabled }

	# Subjects-enabled building types (bool field slot 2)
	cmm_build_list_bool_list = { setting = lsq__auto_build_list field_slot = 2 list_name = lsq_auto_build_building_types_in_subjects_enabled }

	# Min control by building type (numeric field slot 3, CMM 0-100 → 0-1)
	cmm_build_list_field_map = { setting = lsq__auto_build_list field_slot = 3 map_name = lsq_min_control_by_building_type multiply = 0.01 }

	# Min discount by building type (numeric field slot 4, CMM -33..33 → -0.33..0.33)
	cmm_build_list_field_map = { setting = lsq__auto_build_list field_slot = 4 map_name = lsq_min_discount_by_building_type multiply = 0.01 }

	# Min RGOs by building type (numeric field slot 5, direct values)
	cmm_build_list_field_map = { setting = lsq__auto_build_list field_slot = 5 map_name = lsq_min_rgos_by_building_type multiply = 1 }
}

# Benign warning suppression for variables/flags set or used at runtime
# via GUI scripted effects, AddScope, variable maps, or on_actions.
_lsq_suppress_warnings = {
	if = {
		limit = { always = no }
		cmf_suppress = { v = lsq_building_type_processed }
		cmf_suppress = { v = auto_expand_registered_buildings }
		cmf_suppress = { v = lsq_run_auto_expand_rgo }
		cmf_suppress = { v = every_building_type_list }
		cmf_suppress = { v = every_building_type_count }
		cmf_suppress = { v = lsq_hidden_window_initialized }
		cmf_suppress = { v = lsq_auto_expand_rgo_off_in_all_locations }
		cmf_suppress = { v = every_non_standard_building_type }
		cmf_suppress = { v = skip }
		cmf_suppress = { v = set_off }
		# CMM-derived lookup maps (set at runtime by cmm_build_list_* utilities)
		cmf_suppress = { v = lsq_min_control_by_building_type }
		cmf_suppress = { v = lsq_min_discount_by_building_type }
		cmf_suppress = { v = lsq_min_rgos_by_building_type }
		# Legacy individual vars (now read from lsq_min_rgos_by_building_type map, but old triggers still reference)
		cmf_suppress = { v = lsq_required_rgo_levels_before_sawmill }
		cmf_suppress = { v = lsq_required_rgo_levels_before_windmill }
		cmf_suppress = { v = lsq_required_rgo_levels_before_local_smelters }
	}
}
