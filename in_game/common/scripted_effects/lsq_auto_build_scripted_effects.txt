# Root is the country. Expects scope:drop_target_bt and scope:dragged_bt (saved in chain). Reorders lsq_auto_build_building_types_list so dragged is before drop target.
lsq_auto_build_reorder_list_effect = {
  set_local_variable = { name = i value = 0 }
  # debug_log = "Reordering list"
  every_in_list = {
    variable = lsq_auto_build_building_types_list
    save_scope_as = current_bt
    # debug_log = "Adding building type [THIS.GetBuildingType.GetNameWithNoTooltip] to temp list"
    scope:country = { add_to_variable_list = { name = lsq_auto_build_order_temp target = scope:current_bt } }
  }
  clear_variable_list = lsq_auto_build_building_types_list
  set_local_variable = { name = done value = 0 }
  every_in_list = {
    variable = lsq_auto_build_order_temp
    save_scope_as = current_bt
    if = { limit = { NOT = { this = scope:building_type } }
      if = { limit = { local_var:i = local_var:lsq_auto_build_move_target_index }
        # debug_log = "Adding building type [THIS.GetBuildingType.GetNameWithNoTooltip] with moving building type [SCOPE.sBuildingType('building_type').GetNameWithNoTooltip]"
        scope:country = { add_to_variable_list = { name = lsq_auto_build_building_types_list target = scope:building_type } }
        scope:country = { add_to_variable_list = { name = lsq_auto_build_building_types_list target = scope:current_bt } }
        set_local_variable = { name = done value = 1 }
      } else = {
        # debug_log = "Adding building type [THIS.GetBuildingType.GetNameWithNoTooltip]"
        scope:country = { add_to_variable_list = { name = lsq_auto_build_building_types_list target = scope:current_bt } }
      }
      change_local_variable = { name = i add = 1 }
    }
  }
  if = { limit = { local_var:done = 0 }
    # debug_log = "Adding building type [SCOPE.sBuildingType('building_type').GetNameWithNoTooltip] to end of list"
    scope:country = { add_to_variable_list = { name = lsq_auto_build_building_types_list target = scope:building_type } }
  }
  clear_variable_list = lsq_auto_build_order_temp
}

# Root is the country. Expects scope:building_type (the row to move). Moves it one position earlier in the list.
lsq_auto_build_move_building_type_up_effect = {
  # debug_log = "Moving building type up"
  set_local_variable = { name = lsq_auto_build_move_index value = 0 }
  set_local_variable = { name = lsq_auto_build_move_done value = 0 }
  every_in_list = {
    variable = lsq_auto_build_building_types_list
    save_scope_as = current_bt
    # debug_log = "Checking building type [THIS.GetBuildingType.GetNameWithNoTooltip] for matching [SCOPE.sBuildingType('building_type').GetNameWithNoTooltip]"
    if = { limit = { this = scope:building_type }
      scope:country = {
        if = { limit = { AND = { local_var:lsq_auto_build_move_index >= 1 local_var:lsq_auto_build_move_done = 0 } }
          set_local_variable = { name = lsq_auto_build_move_target_index value = { value = local_var:lsq_auto_build_move_index subtract = 1 } }
          lsq_auto_build_reorder_list_effect = {}
          set_local_variable = { name = lsq_auto_build_move_done value = 1 }
        }
      }
    }
    change_local_variable = { name = lsq_auto_build_move_index add = 1 }
    save_scope_as = previous_bt
  }
}

# Root is the country. Expects scope:building_type (the row to move). Moves it one position later in the list.
lsq_auto_build_move_building_type_down_effect = {
  # debug_log = "Moving building type down"
  set_local_variable = { name = lsq_auto_build_move_index value = 0 }
  set_local_variable = { name = lsq_auto_build_move_done value = 0 }
  every_in_list = {
    variable = lsq_auto_build_building_types_list
    save_scope_as = current_bt
    # debug_log = "Checking building type [THIS.GetBuildingType.GetNameWithNoTooltip] for matching [SCOPE.sBuildingType('building_type').GetNameWithNoTooltip]"
    scope:country = {
      if = { limit = { AND = { scope:current_bt = scope:building_type local_var:lsq_auto_build_move_done = 0 } }
        set_local_variable = { name = lsq_auto_build_move_target_index value = { value = local_var:lsq_auto_build_move_index add = 1 } }
        lsq_auto_build_reorder_list_effect = {}
        set_local_variable = { name = lsq_auto_build_move_done value = 1 }
      }
      change_local_variable = { name = lsq_auto_build_move_index add = 1 }
    }
    save_scope_as = previous_bt
  }
}

# Root is the country
lsq_auto_build_init_variables = {
  save_scope_as = country
  # clear_variable_list = lsq_auto_build_building_types_list
  if = { limit = { NOT = { has_variable_list = lsq_auto_build_building_types_list } }
    lsq_reset_auto_build_priorities_to_default_effect = {}
  }
  if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_buildings } }
    set_variable = { name = lsq_minimum_gold_for_buildings value = 50 }
  }
  if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_rgos } }
    set_variable = { name = lsq_minimum_gold_for_rgos value = 50 }
  }
  if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_auto_build } }
    set_variable = { name = lsq_minimum_gold_for_auto_build value = 50 }
  }
  if = { limit = { NOT = { has_variable = lsq_required_rgo_levels_before_sawmill } }
    set_variable = { name = lsq_required_rgo_levels_before_sawmill value = 5 }
  }
  if = { limit = { NOT = { has_variable = lsq_required_rgo_levels_before_windmill } }
    set_variable = { name = lsq_required_rgo_levels_before_windmill value = 5 }
  }
  if = { limit = { NOT = { has_variable = lsq_required_rgo_levels_before_local_smelters } }
    set_variable = { name = lsq_required_rgo_levels_before_local_smelters value = 5 }
  }

  if = { limit = { NOT = { has_variable_map = lsq_minimum_control_for_auto_build } }
    every_in_list = {
      variable = lsq_auto_build_building_types_list
      save_scope_as = building_type
      scope:country = {
        if = { limit = { OR = { NOT = { has_variable_map = lsq_minimum_control_for_auto_build } NOT = { is_key_in_variable_map = { name = lsq_minimum_control_for_auto_build target = scope:building_type } } } }
          # Default to 0 for temple and minting office as they increase control so best to build them everywhere regardless of control
          if = { limit = { OR = { scope:building_type = building_type:temple scope:building_type = building_type:minting_office } }
            add_to_variable_map = { name = lsq_minimum_control_for_auto_build key = scope:building_type value = 0 }
          } else = {
            add_to_variable_map = { name = lsq_minimum_control_for_auto_build key = scope:building_type value = 0.5 }
          }
        }
      }
    }
  }

  lsq_auto_build_init_locations_lists = {}
}

# Root is the country
lsq_auto_build_init_locations_lists = {
  save_scope_as = country
  # Make list of all directly owned locations
  # Also, add to the beginning of the all including subjects list, the sorted list of directly owned locations
  clear_variable_list = lsq_all_controlled_locations_list
  clear_variable_list = lsq_all_locations_including_subjects_list
  ordered_owned_location = {
    order_by = max_control
    save_scope_as = location
    max = lsq_max_guess_at_number_of_locations
    check_range_bounds = no
    scope:country = { add_to_variable_list = { name = lsq_all_controlled_locations_list target = scope:location } }
    scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list target = scope:location } }
    # debug_log = "Adding location [THIS.GetLocation.GetNameWithNoTooltip] to lsq_all_controlled_locations_list with max control [THIS.GetLocation.GetMaxControl]"
  }

  # Make unsorted list of locations of all subjects and below
  clear_variable_list = lsq_all_locations_including_subjects_list_temp
  every_subject_or_below = {
    every_owned_location = {
      save_scope_as = location
      scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list_temp target = scope:location } }
    }
  }

  # Sort and add the subject locations to the end of the list
  ordered_in_list = {
    variable = lsq_all_locations_including_subjects_list_temp
    order_by = max_control
    max = lsq_max_guess_at_number_of_locations
    check_range_bounds = no
    save_scope_as = location
    scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list target = scope:location } }
  }
  clear_variable_list = lsq_all_locations_including_subjects_list_temp
}
