# Root is the country
lsq_auto_build_init_variables = {
  clear_variable_list = lsq_auto_build_building_types_list
  add_to_variable_list = { name = lsq_auto_build_building_types_list target = building_type:temple }
  add_to_variable_list = { name = lsq_auto_build_building_types_list target = building_type:library }
  add_to_variable_list = { name = lsq_auto_build_building_types_list target = building_type:settlement_building }
  add_to_variable_list = { name = lsq_auto_build_building_types_list target = building_type:bridge_infrastructure }
  add_to_variable_list = { name = lsq_auto_build_building_types_list target = building_type:university }

  lsq_auto_build_init_locations_lists = {}
}

# Root is the country
lsq_auto_build_init_locations_lists = {
  save_scope_as = country
  # Make list of all directly controlled locations
  # Also, add to the beginning of the all including subjects list, the sorted list of directly controlled locations
  clear_variable_list = lsq_all_controlled_locations_list
  clear_variable_list = lsq_all_locations_including_subjects_list
  ordered_controlled_location = {
    order_by = max_control
    save_scope_as = location
    max = lsq_max_guess_at_number_of_locations
    check_range_bounds = no
    scope:country = { add_to_variable_list = { name = lsq_all_controlled_locations_list target = scope:location } }
    scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list target = scope:location } }
    # debug_log = "Adding location [THIS.GetLocation.GetNameWithNoTooltip] to lsq_all_controlled_locations_list with max control [THIS.GetLocation.GetMaxControl]"
  }

  # Make unsorted list of locations of all subjects and below
  clear_variable_list = lsq_all_locations_including_subjects_list_temp
  every_subject_or_below = {
    every_controlled_location = {
      save_scope_as = location
      scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list_temp target = scope:location } }
    }
  }

  # Sort and add the subject locations to the end of the list
  ordered_in_list = {
    variable = lsq_all_locations_including_subjects_list_temp
    order_by = max_control
    max = lsq_max_guess_at_number_of_locations
    check_range_bounds = no
    save_scope_as = location
    scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list target = scope:location } }
  }
  clear_variable_list = lsq_all_locations_including_subjects_list_temp
}
