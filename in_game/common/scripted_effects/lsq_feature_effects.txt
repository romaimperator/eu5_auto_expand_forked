# Feature effects for the unified priority dispatcher.
# Each effect is called by the dispatcher in lsq_on_action.txt when
# the corresponding feature is enabled in the priority list.
# All effects expect scope:country to be set.

# ─── Auto Expand Buildings ───
# Expands multi-level buildings that are profitable and have pops available.
# Flags obsolete buildings for GUI-driven upgrade processing.
lsq_run_auto_expand_buildings = {
	# Clear the GUI upgrade locations list so it's rebuilt fresh each tick
	clear_variable_list = lsq_gui_upgrade_locations
	if = {
		limit = { has_variable_list = lsq_mass_auto_expand_building_types }
		# ── Mass mode: iterate all existing buildings at all owned locations ──
		ordered_owned_location = {
			order_by = max_control
			max = lsq_max_guess_at_number_of_locations
			check_range_bounds = no
			limit = { NOT = { integration_level = conquered } }
			save_scope_as = location
			ordered_buildings_in_location = {
				order_by = building_index
				max = 100
				check_range_bounds = no
				save_scope_as = building
				building_type = { save_scope_as = building_type }
				if = {
					limit = {
						# Building type is mass-toggled for this country
						scope:country = { is_target_in_variable_list = { name = lsq_mass_auto_expand_building_types target = scope:building_type } }
						# Not excluded by the player at this location
						NOT = { scope:location = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:building_type } } }
					}
					if = {
						limit = { lsq_is_building_type_obsolete = no }
						if = {
							limit = {
								scope:building = {
									lsq_building_can_auto_expand_other_than_checking_for_upgrades = yes
									is_at_max_level = no
									lsq_is_not_under_construction_and_gold_is_available = yes
								}
							}
							scope:location = { construct_building = { building_type = scope:building_type } }
						}
					} else_if = {
						limit = {
							scope:location = { lsq_is_building_type_upgradeable_and_can_be_upgraded_by_owner = yes }
						}
						lsq_clear_finished_upgrade_in_progress = yes
						if = {
							limit = {
								scope:building = {
									lsq_building_can_auto_expand_for_upgrade = yes
									lsq_is_not_under_construction = yes
								}
								lsq_auto_expand_check_minimum_gold_reserve = yes
								NOT = { scope:location = { is_target_in_variable_list = { name = lsq_upgrade_in_progress target = scope:building_type } } }
							}
							scope:location = {
								if = {
									limit = { NOT = { is_target_in_variable_list = { name = lsq_gui_upgrade_building_types target = scope:building_type } } }
									add_to_variable_list = { name = lsq_gui_upgrade_building_types target = scope:building_type }
								}
							}
							scope:country = {
								if = {
									limit = { NOT = { is_target_in_variable_list = { name = lsq_gui_upgrade_locations target = scope:location } } }
									add_to_variable_list = { name = lsq_gui_upgrade_locations target = scope:location }
								}
							}
						}
					}
				}
			}
			# ── Mass mode: construct missing buildings from zero levels ──
			scope:country = {
				every_in_list = {
					variable = lsq_mass_auto_expand_building_types
					save_scope_as = building_type
					if = {
						limit = {
							NOT = { scope:location = { has_building = scope:building_type } }
							NOT = { scope:location = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = scope:building_type } } }
						}
						# Clean up in-progress upgrade tracking if the old building type is gone (upgrade completed)
						if = {
							limit = { scope:location = { is_target_in_variable_list = { name = lsq_upgrade_in_progress target = scope:building_type } } }
							scope:location = { remove_list_variable = { name = lsq_upgrade_in_progress target = scope:building_type } }
						}
						if = {
							limit = { lsq_is_building_type_obsolete = no }
							if = {
								limit = { lsq_location_can_auto_expand_from_zero_levels = yes }
								scope:location = { construct_building = { building_type = scope:building_type } }
							}
						} else_if = {
							limit = {
								scope:location = { lsq_is_building_type_upgradeable_and_can_be_upgraded_by_owner = yes }
								}
							lsq_clear_finished_upgrade_in_progress = yes
							if = {
								limit = {
									lsq_location_can_auto_expand_from_zero_levels_for_upgrade = yes
									NOT = { scope:location = { is_target_in_variable_list = { name = lsq_upgrade_in_progress target = scope:building_type } } }
								}
								scope:location = {
									if = {
										limit = { NOT = { is_target_in_variable_list = { name = lsq_gui_upgrade_building_types target = scope:building_type } } }
										add_to_variable_list = { name = lsq_gui_upgrade_building_types target = scope:building_type }
									}
								}
								scope:country = {
									if = {
										limit = { NOT = { is_target_in_variable_list = { name = lsq_gui_upgrade_locations target = scope:location } } }
										add_to_variable_list = { name = lsq_gui_upgrade_locations target = scope:location }
									}
								}
							}
						}
					}
				}
			}
		}
	}
	# ── Normal mode: iterate per-location registered building types ──
	# Runs unconditionally — per-location registrations are processed alongside mass mode.
	ordered_owned_location = {
		order_by = max_control
		max = lsq_max_guess_at_number_of_locations
		check_range_bounds = no
		limit = { NOT = { integration_level = conquered } has_variable_list = lsq_auto_expand_registered_building_types }
		save_scope_as = location
		every_in_list = {
			variable = lsq_auto_expand_registered_building_types
			save_scope_as = building_type
			if = {
				limit = { scope:location = { has_building = scope:building_type } }
				scope:location = { "building(scope:building_type|scope:country)" = { save_scope_as = building } }
				if = {
					limit = { lsq_is_building_type_obsolete = no }
					if = {
						limit = {
							scope:building = {
								lsq_building_can_auto_expand_other_than_checking_for_upgrades = yes
								is_at_max_level = no
								lsq_is_not_under_construction_and_gold_is_available = yes
							}
						}
						scope:location = { construct_building = { building_type = scope:building_type } }
					}
				} else_if = {
					limit = {
						scope:location = { lsq_is_building_type_upgradeable_and_can_be_upgraded_by_owner = yes }
					}
					lsq_clear_finished_upgrade_in_progress = yes
					if = {
						limit = {
							scope:building = {
								lsq_building_can_auto_expand_for_upgrade = yes
								lsq_is_not_under_construction = yes
							}
							lsq_auto_expand_check_minimum_gold_reserve = yes
							NOT = { scope:location = { is_target_in_variable_list = { name = lsq_upgrade_in_progress target = scope:building_type } } }
						}
						scope:location = {
							if = {
								limit = { NOT = { is_target_in_variable_list = { name = lsq_gui_upgrade_building_types target = scope:building_type } } }
								add_to_variable_list = { name = lsq_gui_upgrade_building_types target = scope:building_type }
							}
						}
						scope:country = {
							if = {
								limit = { NOT = { is_target_in_variable_list = { name = lsq_gui_upgrade_locations target = scope:location } } }
								add_to_variable_list = { name = lsq_gui_upgrade_locations target = scope:location }
							}
						}
					}
				}
			} else = {
				# Building has no levels — not yet queued for construction
				# Clean up in-progress upgrade tracking if the old building type is gone (upgrade completed)
				if = {
					limit = { scope:location = { is_target_in_variable_list = { name = lsq_upgrade_in_progress target = scope:building_type } } }
					scope:location = { remove_list_variable = { name = lsq_upgrade_in_progress target = scope:building_type } }
				}
				if = {
					limit = { lsq_is_building_type_obsolete = no }
					if = {
						limit = { lsq_location_can_auto_expand_from_zero_levels = yes }
						scope:location = { construct_building = { building_type = scope:building_type } }
					}
				} else_if = {
					limit = { scope:location = { lsq_is_building_type_upgradeable_and_can_be_upgraded_by_owner = yes } }
					lsq_clear_finished_upgrade_in_progress = yes
					if = {
						limit = {
							lsq_location_can_auto_expand_from_zero_levels_for_upgrade = yes
							NOT = { scope:location = { is_target_in_variable_list = { name = lsq_upgrade_in_progress target = scope:building_type } } }
						}
						scope:location = {
							if = {
								limit = { NOT = { is_target_in_variable_list = { name = lsq_gui_upgrade_building_types target = scope:building_type } } }
								add_to_variable_list = { name = lsq_gui_upgrade_building_types target = scope:building_type }
							}
						}
						scope:country = {
							if = {
								limit = { NOT = { is_target_in_variable_list = { name = lsq_gui_upgrade_locations target = scope:location } } }
								add_to_variable_list = { name = lsq_gui_upgrade_locations target = scope:location }
							}
						}
					}
				}
			}
		}
	}
}

# ─── Auto Expand RGOs ───
# Expands RGO levels in locations with auto_expand_rgo enabled.
# Single monthly pass (no daily spread).
lsq_run_auto_expand_rgos = {
	if = {
		limit = { exists = var:lsq_mass_auto_expand_rgo }
		# Mass mode: expand all non-excluded locations
		ordered_owned_location = {
			order_by = max_control
			max = lsq_max_guess_at_number_of_locations
			check_range_bounds = no
			limit = { NOT = { exists = var:auto_expand_rgo_excluded } NOT = { integration_level = conquered } }
			save_scope_as = location
			if = {
				limit = { lsq_should_rgo_auto_expand = yes }
				construct_rgo_upgrade = { }
			}
		}
	} else = {
		# Normal mode: expand only locations with auto_expand_rgo enabled
		ordered_owned_location = {
			order_by = max_control
			max = lsq_max_guess_at_number_of_locations
			check_range_bounds = no
			limit = { exists = var:auto_expand_rgo NOT = { integration_level = conquered } }
			save_scope_as = location
			if = {
				limit = { lsq_should_rgo_auto_expand = yes }
				construct_rgo_upgrade = { }
			}
		}
	}
}

# Clears lsq_upgrade_in_progress for scope:building_type at scope:location when the
# previously queued upgrade construction has finished or was cancelled. This allows
# the next monthly pulse to queue another upgrade for the same building type.
# Expects: scope:building_type, scope:location, scope:country
lsq_clear_finished_upgrade_in_progress = {
	if = {
		limit = {
			scope:location = { is_target_in_variable_list = { name = lsq_upgrade_in_progress target = scope:building_type } }
		}
		if = {
			limit = { is_key_in_global_variable_map = { name = lsq_building_type_to_upgrade_type target = scope:building_type } }
			"global_variable_map(lsq_building_type_to_upgrade_type|scope:building_type)" = { save_temporary_scope_as = lsq_upgrade_target }
			if = {
				limit = {
					trigger_if = {
						limit = { scope:location = { has_building = scope:lsq_upgrade_target } }
						# Upgrade target exists but has no levels under construction — construction finished
						scope:location = { "building(scope:lsq_upgrade_target|scope:country)" = { building_levels_under_construction = 0 } }
					}
					trigger_else = {
						# Upgrade target doesn't exist — construction was cancelled
						always = yes
					}
				}
				scope:location = { remove_list_variable = { name = lsq_upgrade_in_progress target = scope:building_type } }
			}
		} else = {
			# No upgrade mapping found — stale entry, clean up
			scope:location = { remove_list_variable = { name = lsq_upgrade_in_progress target = scope:building_type } }
		}
	}
}

# ─── Auto Build ───
# Auto-constructs one-per-location buildings in priority order.
lsq_run_auto_build = {
	lsq_reset_locations_lists = { }
	every_in_list = {
		variable = lsq_auto_build_building_types_list
		save_scope_as = building_type
		scope:country = {
			if = {
				limit = { is_target_in_variable_list = { name = lsq_auto_build_building_types_enabled target = scope:building_type } }
				# Save in_subjects flag before resolution overwrites scope:building_type
				if = { limit = { is_target_in_variable_list = { name = lsq_auto_build_building_types_in_subjects_enabled target = scope:building_type } }
					set_local_variable = { name = lsq_auto_build_include_subjects value = 1 }
				} else = {
					set_local_variable = { name = lsq_auto_build_include_subjects value = 0 }
				}
				# Resolve obsolete building type to current non-obsolete upgrade (e.g. dock → dry_dock)
				lsq_auto_build_resolve_building_type = yes
				if = {
					limit = { local_var:lsq_auto_build_include_subjects = 0 }
					ordered_owned_location = {
						order_by = max_control
						max = lsq_max_guess_at_number_of_locations
						check_range_bounds = no
						limit = { NOT = { integration_level = conquered } }
						save_scope_as = location
						if = {
							limit = { AND = { lsq_get_open_build_queue_slots > 0 lsq_location_can_auto_build = yes } }
							construct_building = { building_type = scope:building_type }
						}
					}
				} else = {
					every_in_list = {
						variable = lsq_all_locations_including_subjects_list
						save_scope_as = location
						if = {
							limit = { AND = { NOT = { integration_level = conquered } lsq_get_open_build_queue_slots > 0 lsq_location_can_auto_build = yes } }
							if = {
								limit = { owner = scope:country }
								construct_building = { building_type = scope:building_type }
							} else = {
								construct_building = {
									building_type = scope:building_type
									owner = scope:country
									payer = scope:country
								}
							}
						}
					}
				}
			}
		}
	}
	lsq_reset_locations_lists = { }
}

# ─── Auto Food ───
# Auto-builds food RGOs and related buildings in priority order.
# Single monthly pass (no daily spread).
lsq_run_auto_food = {
	lsq_update_food_data_columns = yes
	if = {
		limit = { has_variable_list = lsq_auto_food_rgo_types_list }
		every_in_list = {
			variable = lsq_auto_food_rgo_types_list
			save_scope_as = food_good
			scope:country = {
				if = {
					limit = { is_target_in_variable_list = { name = lsq_auto_food_rgo_types_enabled target = scope:food_good } }
					lsq_food_auto_build_for_rgo_type = yes
				}
			}
		}
	}
}

# ─── Auto Upgrade Towns to Cities ───
# Upgrades qualifying towns to cities.
lsq_run_auto_upgrade_towns_to_cities = {
	ordered_owned_location = {
		order_by = max_control
		max = lsq_max_guess_at_number_of_locations
		check_range_bounds = no
		save_scope_as = location
		limit = { NOT = { integration_level = conquered } lsq_location_can_mass_upgrade_to_city = yes }
		if = {
			limit = { lsq_check_country_gold_above = { threshold = lsq_mass_upgrade_town_to_city_gold_needed } }
			scope:location = { construct_building = { building_type = building_type:lsq_upgrade_town_to_city_building } }
		}
	}
}
