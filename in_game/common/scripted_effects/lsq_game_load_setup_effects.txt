# Populates the global list of building types for classification by the hidden window GUI.
# Called from on_game_start_after_lobby and on_game_load_after_lobby (once per lobby, not per player).
# Stateless: always rebuilds from scratch so no version tracking or migration is needed.
lsq_populate_building_types_to_process = {
  clear_global_variable_list = lsq_building_types_to_process
  set_global_variable = { name = lsq_building_types_to_process_count value = 0 }
  clear_global_variable_map = lsq_building_type_processed
  every_building_type = {
    add_to_global_variable_list = { name = lsq_building_types_to_process target = this }
    change_global_variable = { name = lsq_building_types_to_process_count add = 1 }
  }
  # Trigger building classification in the hidden window GUI
  set_global_variable = { name = lsq_process_buildings value = yes }
}

# Expects scopes: country
lsq_check_that_country_variables_are_set = {
  scope:country = {
    # Reset the location lists every time we load the game
    if = { limit = { OR = { NOT = { has_variable_list = lsq_all_controlled_locations_list } NOT = { has_variable_list = lsq_all_locations_including_subjects_list } } }
      lsq_reset_locations_lists = {}
    }

    # Simple settings (CMM aliases — initialized by CMM registration, defaults here for non-CMF fallback)
    if = { limit = { NOT = { has_variable = lsq_minimum_gold } }
      set_variable = { name = lsq_minimum_gold value = 50 }
    }
    if = { limit = { NOT = { has_variable = lsq_allowed_building_overbuild_levels } }
      set_variable = { name = lsq_allowed_building_overbuild_levels value = 0 }
    }

  }
}

# Expects scopes: country
lsq_reset_locations_lists = {
  scope:country = {
    # Make list of all directly owned locations
    # Also, add to the beginning of the all including subjects list, the sorted list of directly owned locations
    clear_variable_list = lsq_all_controlled_locations_list
    clear_variable_list = lsq_all_locations_including_subjects_list
    ordered_owned_location = {
      order_by = max_control
      save_scope_as = location
      max = lsq_max_guess_at_number_of_locations
      check_range_bounds = no
      scope:country = { add_to_variable_list = { name = lsq_all_controlled_locations_list target = scope:location } }
      scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list target = scope:location } }
      # debug_log = "Adding location [THIS.GetLocation.GetNameWithNoTooltip] to lsq_all_controlled_locations_list with max control [THIS.GetLocation.GetMaxControl]"
    }

    # Make unsorted list of locations of all subjects and below
    clear_variable_list = lsq_all_locations_including_subjects_list_temp
    every_subject_or_below = {
      every_owned_location = {
        save_scope_as = location
        scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list_temp target = scope:location } }
      }
    }

    # Sort and add the subject locations to the end of the list
    ordered_in_list = {
      variable = lsq_all_locations_including_subjects_list_temp
      order_by = max_control
      max = lsq_max_guess_at_number_of_locations
      check_range_bounds = no
      save_scope_as = location
      scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list target = scope:location } }
    }
    clear_variable_list = lsq_all_locations_including_subjects_list_temp

    # Setup building-specific lists
    lsq_process_building_specific_list = { building_type_text = temple } # buiding_type_index = 0 }
    lsq_process_building_specific_list = { building_type_text = library } # buiding_type_index = 1 }
    lsq_process_building_specific_list = { building_type_text = settlement_building } # buiding_type_index = 2 }
    lsq_process_building_specific_list = { building_type_text = bridge_infrastructure } # buiding_type_index = 3 }
    lsq_process_building_specific_list = { building_type_text = pound_lock_canal_infrastructure } # buiding_type_index = 4 }
    lsq_process_building_specific_list = { building_type_text = university } # buiding_type_index = 5 }
    lsq_process_building_specific_list = { building_type_text = hospital } # buiding_type_index = 6 }
    lsq_process_building_specific_list = { building_type_text = minting_office } # buiding_type_index = 7 }
    lsq_process_building_specific_list = { building_type_text = counting_house } # buiding_type_index = 8 }
    lsq_process_building_specific_list = { building_type_text = wharf } # buiding_type_index = 9 }
    lsq_process_building_specific_list = { building_type_text = market_village } # buiding_type_index = 10 }
    lsq_process_building_specific_list = { building_type_text = sawmill } # buiding_type_index = 11 }
    lsq_process_building_specific_list = { building_type_text = windmill } # buiding_type_index = 12 }
    lsq_process_building_specific_list = { building_type_text = local_smelters } # buiding_type_index = 13 }
    lsq_process_building_specific_list = { building_type_text = protected_harbor } # buiding_type_index = 14 }
    lsq_process_building_specific_list = { building_type_text = fortress_granary } # buiding_type_index = 15 }

# Expects scopes: country
lsq_process_building_specific_list = {
  # debug_log = "LSQ: Processing building specific list for  $building_type_text$"
  scope:country = {
    clear_variable_list = lsq_all_controlled_locations_list_allowed_to_build_$building_type_text$
    set_variable = { name = lsq_all_controlled_locations_size_will_auto_build_$building_type_text$ value = 0 }
    every_in_list = {
      variable = lsq_all_controlled_locations_list
      save_scope_as = location
      if = { limit = { scope:location = { lsq_location_allowed_to_build_$building_type_text$ = yes } }
        scope:country = { add_to_variable_list = { name = lsq_all_controlled_locations_list_allowed_to_build_$building_type_text$ target = scope:location } }
        if = { limit = { scope:location = { lsq_location_will_auto_build_$building_type_text$ = yes } }
          # debug_log = "LSQ: Adding location will auto build [THIS.GetLocation.GetNameWithNoTooltip]"
          scope:country = { change_variable = { name = lsq_all_controlled_locations_size_will_auto_build_$building_type_text$ add = 1 } }
        }
      }
    }
    clear_variable_list = lsq_all_locations_including_subjects_list_allowed_to_build_$building_type_text$
    set_variable = { name = lsq_all_locations_including_subjects_size_will_auto_build_$building_type_text$ value = 0 }
    every_in_list = {
      variable = lsq_all_locations_including_subjects_list
      save_scope_as = location
      if = { limit = { scope:location = { lsq_location_allowed_to_build_$building_type_text$ = yes } }
        scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list_allowed_to_build_$building_type_text$ target = scope:location } }
        if = { limit = { scope:location = { lsq_location_will_auto_build_$building_type_text$ = yes } }
          scope:country = { change_variable = { name = lsq_all_locations_including_subjects_size_will_auto_build_$building_type_text$ add = 1 } }
        }
      }
    }
  }
}
