# Expects scopes: country
lsq_check_that_country_variables_are_set = {
  scope:country = {
    # # Setup/Refresh the list of all building types
    # if = { limit = { NOT = { has_global_variable_list = every_building_type_list } }
    #   lsq_setup_every_building_type_list = {}
    # }

    # Setup the building type type maps
    if = { limit = { OR = { NOT = { has_global_variable_map = lsq_building_type_to_is_production_building } NOT = { has_global_variable_map = lsq_building_type_to_is_non_production_multi_level_building } NOT = { has_global_variable_map = lsq_building_type_to_is_one_per_location_building } NOT = { has_global_variable_map = lsq_building_type_to_is_fort } } }
      lsq_populate_building_type_type_maps = {}
    }

    # Reset the location lists every time we load the game
    if = { limit = { OR = { NOT = { has_variable_list = lsq_all_controlled_locations_list } NOT = { has_variable_list = lsq_all_locations_including_subjects_list } } }
      lsq_reset_locations_lists = {}
    }

    if = { limit = { NOT = { has_variable_list = lsq_auto_build_building_types_list } }
      lsq_reset_auto_build_priorities_to_default_effect = {}
    }
    if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_buildings } }
      set_variable = { name = lsq_minimum_gold_for_buildings value = 50 }
    }
    if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_rgos } }
      set_variable = { name = lsq_minimum_gold_for_rgos value = 50 }
    }
    if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_auto_build } }
      set_variable = { name = lsq_minimum_gold_for_auto_build value = 50 }
    }
    if = { limit = { NOT = { has_variable = lsq_allowed_building_overbuild_levels } }
      set_variable = { name = lsq_allowed_building_overbuild_levels value = 0 }
    }
    if = { limit = { NOT = { has_variable = lsq_required_rgo_levels_before_sawmill } }
      set_variable = { name = lsq_required_rgo_levels_before_sawmill value = 5 }
    }
    if = { limit = { NOT = { has_variable = lsq_required_rgo_levels_before_windmill } }
      set_variable = { name = lsq_required_rgo_levels_before_windmill value = 5 }
    }
    if = { limit = { NOT = { has_variable = lsq_required_rgo_levels_before_local_smelters } }
      set_variable = { name = lsq_required_rgo_levels_before_local_smelters value = 5 }
    }

    if = { limit = { NOT = { has_variable_map = lsq_minimum_control_for_auto_build } }
      every_in_list = {
        variable = lsq_auto_build_building_types_list
        save_scope_as = building_type
        scope:country = {
          if = { limit = { OR = { NOT = { has_variable_map = lsq_minimum_control_for_auto_build } NOT = { is_key_in_variable_map = { name = lsq_minimum_control_for_auto_build target = scope:building_type } } } }
            # Default to 0 for temple and minting office as they increase control so best to build them everywhere regardless of control, and settlements as we want to build them everywhere for the population growth
            if = { limit = { OR = { scope:building_type = building_type:temple scope:building_type = building_type:minting_office scope:building_type = building_type:settlement_building } }
              add_to_variable_map = { name = lsq_minimum_control_for_auto_build key = scope:building_type value = 0 }
            } else = {
              add_to_variable_map = { name = lsq_minimum_control_for_auto_build key = scope:building_type value = 0.5 }
            }
          }
        }
      }
    }

    if = { limit = { NOT = { has_variable_map = lsq_minimum_market_price_impact_for_auto_build } }
      every_in_list = {
        variable = lsq_auto_build_building_types_list
        save_scope_as = building_type
        scope:country = {
          if = { limit = { OR = { NOT = { has_variable_map = lsq_minimum_market_price_impact_for_auto_build } NOT = { is_key_in_variable_map = { name = lsq_minimum_market_price_impact_for_auto_build target = scope:building_type } } } }
            # Default to -0.33 for all building types so by default we will build only with the max discount
            add_to_variable_map = { name = lsq_minimum_market_price_impact_for_auto_build key = scope:building_type value = -0.33 }
          }
        }
      }
    }
  }
}

# Expects scopes: old_country, new_country
lsq_move_data_from_old_country_to_new_country = {
  # FYI, we can't remove what buildings and RGOs were enabled for auto-expansion without breaking the game when someone uses the world map to change countries since it counts as a tag switch.
  debug_log = "LSQ: Moving data from old country to new country"

  scope:new_country = {
    # Copy the mod version
    lsq_copy_mod_version_from_old_country_to_new_country = {}

    save_temporary_scope_as = country
    # Don't bother copying but reset them to get the new locations for the new country
    lsq_reset_locations_lists = {}

    scope:old_country = {
      # Copy the order of auto-build building types
      every_in_list = {
        variable = lsq_auto_build_building_types_list
        save_scope_as = building_type
        scope:new_country = {
          add_to_variable_list = { name = lsq_auto_build_building_types_list target = scope:building_type }
        }
      }

      # Copy the enabled auto-build building types
      every_in_list = {
        variable = lsq_auto_build_building_types_enabled
        save_scope_as = building_type
        scope:new_country = {
          add_to_variable_list = { name = lsq_auto_build_building_types_enabled target = scope:building_type }
        }
      }

      # Copy the in subjects auto-build building types
      every_in_list = {
        variable = lsq_auto_build_building_types_in_subjects_enabled
        save_scope_as = building_type
        scope:new_country = {
          add_to_variable_list = { name = lsq_auto_build_building_types_in_subjects_enabled target = scope:building_type }
        }
      }
    }

    # Copy the basic settings
    if = { limit = { scope:old_country = { has_variable = lsq_settings_window_visible } }
      set_variable = { name = lsq_settings_window_visible value = scope:old_country.var:lsq_settings_window_visible }
    }
    if = { limit = { scope:old_country = { has_variable = lsq_pause_auto_expand } }
      set_variable = { name = lsq_pause_auto_expand value = scope:old_country.var:lsq_pause_auto_expand }
    }
    if = { limit = { scope:old_country = { has_variable = lsq_should_not_upgrade_buildings } }
      set_variable = { name = lsq_should_not_upgrade_buildings value = scope:old_country.var:lsq_should_not_upgrade_buildings }
    }
    if = { limit = { scope:old_country = { has_variable = lsq_minimum_gold_for_buildings } }
      set_variable = { name = lsq_minimum_gold_for_buildings value = scope:old_country.var:lsq_minimum_gold_for_buildings }
    }
    if = { limit = { scope:old_country = { has_variable = lsq_minimum_gold_for_rgos } }
      set_variable = { name = lsq_minimum_gold_for_rgos value = scope:old_country.var:lsq_minimum_gold_for_rgos }
    }
    if = { limit = { scope:old_country = { has_variable = lsq_minimum_gold_for_auto_build } }
      set_variable = { name = lsq_minimum_gold_for_auto_build value = scope:old_country.var:lsq_minimum_gold_for_auto_build }
    }
    if = { limit = { scope:old_country = { has_variable = lsq_allowed_building_overbuild_levels } }
      set_variable = { name = lsq_allowed_building_overbuild_levels value = scope:old_country.var:lsq_allowed_building_overbuild_levels }
    }
    if = { limit = { scope:old_country = { has_variable = lsq_required_rgo_levels_before_sawmill } }
      set_variable = { name = lsq_required_rgo_levels_before_sawmill value = scope:old_country.var:lsq_required_rgo_levels_before_sawmill }
    }
    if = { limit = { scope:old_country = { has_variable = lsq_required_rgo_levels_before_windmill } }
      set_variable = { name = lsq_required_rgo_levels_before_windmill value = scope:old_country.var:lsq_required_rgo_levels_before_windmill }
    }
    if = { limit = { scope:old_country = { has_variable = lsq_required_rgo_levels_before_local_smelters } }
      set_variable = { name = lsq_required_rgo_levels_before_local_smelters value = scope:old_country.var:lsq_required_rgo_levels_before_local_smelters }
    }

    # Copy the minimum control settings
    scope:old_country = {
      every_key_in_variable_map = {
        variable = lsq_minimum_control_for_auto_build
        save_scope_as = building_type_key
        set_local_variable = { name = min_control value = "scope:old_country.variable_map(lsq_minimum_control_for_auto_build|scope:building_type_key)" }
        debug_log = "LSQ: Copying minimum control for auto build for building type [TopScope.sBuildingType('building_type_key').GetNameWithNoTooltip] to [TopScope.GetLocalVariable('min_control').GetValue]"
        scope:new_country = {
          add_to_variable_map = { name = lsq_minimum_control_for_auto_build key = scope:building_type_key value = local_var:min_control }
        }
      }
    }

    # Copy the minimum market_price_impact settings
    scope:old_country = {
      every_key_in_variable_map = {
        variable = lsq_minimum_market_price_impact_for_auto_build
        save_scope_as = building_type_key
        set_local_variable = { name = min_market_price_impact value = "scope:old_country.variable_map(lsq_minimum_market_price_impact_for_auto_build|scope:building_type_key)" }
        debug_log = "LSQ: Copying minimum market_price_impact for auto build for building type [TopScope.sBuildingType('building_type_key').GetNameWithNoTooltip] to [TopScope.GetLocalVariable('min_market_price_impact').GetValue]"
        scope:new_country = {
          add_to_variable_map = { name = lsq_minimum_market_price_impact_for_auto_build key = scope:building_type_key value = local_var:min_market_price_impact }
        }
      }
    }
  }

  # Clear the old country variables
  scope:old_country = {
    clear_variable_list = lsq_all_controlled_locations_list
    clear_variable_list = lsq_all_locations_including_subjects_list
    clear_variable_list = lsq_auto_build_building_types_list
    clear_variable_list = lsq_auto_build_building_types_enabled
    clear_variable_list = lsq_auto_build_building_types_in_subjects_enabled
    remove_variable = lsq_settings_window_visible
    remove_variable = lsq_pause_auto_expand
    remove_variable = lsq_should_not_upgrade_buildings
    remove_variable = lsq_minimum_gold_for_buildings
    remove_variable = lsq_minimum_gold_for_rgos
    remove_variable = lsq_minimum_gold_for_auto_build
    remove_variable = lsq_allowed_building_overbuild_levels
    remove_variable = lsq_required_rgo_levels_before_sawmill
    remove_variable = lsq_required_rgo_levels_before_windmill
    remove_variable = lsq_required_rgo_levels_before_local_smelters
    clear_variable_map = lsq_minimum_control_for_auto_build
    clear_variable_map = lsq_minimum_market_price_impact_for_auto_build
  }
}

# Expects scopes: country
lsq_reset_locations_lists = {
  scope:country = {
    # Make list of all directly owned locations
    # Also, add to the beginning of the all including subjects list, the sorted list of directly owned locations
    clear_variable_list = lsq_all_controlled_locations_list
    clear_variable_list = lsq_all_locations_including_subjects_list
    ordered_owned_location = {
      order_by = max_control
      save_scope_as = location
      max = lsq_max_guess_at_number_of_locations
      check_range_bounds = no
      scope:country = { add_to_variable_list = { name = lsq_all_controlled_locations_list target = scope:location } }
      scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list target = scope:location } }
      # debug_log = "Adding location [THIS.GetLocation.GetNameWithNoTooltip] to lsq_all_controlled_locations_list with max control [THIS.GetLocation.GetMaxControl]"
    }

    # Make unsorted list of locations of all subjects and below
    clear_variable_list = lsq_all_locations_including_subjects_list_temp
    every_subject_or_below = {
      every_owned_location = {
        save_scope_as = location
        scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list_temp target = scope:location } }
      }
    }

    # Sort and add the subject locations to the end of the list
    ordered_in_list = {
      variable = lsq_all_locations_including_subjects_list_temp
      order_by = max_control
      max = lsq_max_guess_at_number_of_locations
      check_range_bounds = no
      save_scope_as = location
      scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list target = scope:location } }
    }
    clear_variable_list = lsq_all_locations_including_subjects_list_temp

    # Setup building-specific lists
    lsq_process_building_specific_list = { building_type_text = temple } # buiding_type_index = 0 }
    lsq_process_building_specific_list = { building_type_text = library } # buiding_type_index = 1 }
    lsq_process_building_specific_list = { building_type_text = settlement_building } # buiding_type_index = 2 }
    lsq_process_building_specific_list = { building_type_text = bridge_infrastructure } # buiding_type_index = 3 }
    lsq_process_building_specific_list = { building_type_text = pound_lock_canal_infrastructure } # buiding_type_index = 4 }
    lsq_process_building_specific_list = { building_type_text = university } # buiding_type_index = 5 }
    lsq_process_building_specific_list = { building_type_text = hospital } # buiding_type_index = 6 }
    lsq_process_building_specific_list = { building_type_text = minting_office } # buiding_type_index = 7 }
    lsq_process_building_specific_list = { building_type_text = counting_house } # buiding_type_index = 8 }
    lsq_process_building_specific_list = { building_type_text = wharf } # buiding_type_index = 9 }
    lsq_process_building_specific_list = { building_type_text = market_village } # buiding_type_index = 10 }
    lsq_process_building_specific_list = { building_type_text = sawmill } # buiding_type_index = 11 }
    lsq_process_building_specific_list = { building_type_text = windmill } # buiding_type_index = 12 }
    lsq_process_building_specific_list = { building_type_text = local_smelters } # buiding_type_index = 13 }
    lsq_process_building_specific_list = { building_type_text = protected_harbor } # buiding_type_index = 14 }
  }
}

# Expects scopes: country
lsq_process_building_specific_list = {
  debug_log = "LSQ: Processing building specific list for  $building_type_text$"
  scope:country = {
    clear_variable_list = lsq_all_controlled_locations_list_allowed_to_build_$building_type_text$
    set_variable = { name = lsq_all_controlled_locations_size_will_auto_build_$building_type_text$ value = 0 }
    every_in_list = {
      variable = lsq_all_controlled_locations_list
      save_scope_as = location
      if = { limit = { scope:location = { lsq_location_allowed_to_build_$building_type_text$ = yes } }
        scope:country = { add_to_variable_list = { name = lsq_all_controlled_locations_list_allowed_to_build_$building_type_text$ target = scope:location } }
        if = { limit = { scope:location = { lsq_location_will_auto_build_$building_type_text$ = yes } }
          # debug_log = "LSQ: Adding location will auto build [THIS.GetLocation.GetNameWithNoTooltip]"
          scope:country = { change_variable = { name = lsq_all_controlled_locations_size_will_auto_build_$building_type_text$ add = 1 } }
        }
      }
    }
    clear_variable_list = lsq_all_locations_including_subjects_list_allowed_to_build_$building_type_text$
    set_variable = { name = lsq_all_locations_including_subjects_size_will_auto_build_$building_type_text$ value = 0 }
    every_in_list = {
      variable = lsq_all_locations_including_subjects_list
      save_scope_as = location
      if = { limit = { scope:location = { lsq_location_allowed_to_build_$building_type_text$ = yes } }
        scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list_allowed_to_build_$building_type_text$ target = scope:location } }
        if = { limit = { scope:location = { lsq_location_will_auto_build_$building_type_text$ = yes } }
          scope:country = { change_variable = { name = lsq_all_locations_including_subjects_size_will_auto_build_$building_type_text$ add = 1 } }
        }
      }
    }
  }
}
