# Expects scopes: country
lsq_check_that_country_variables_are_set = {
  scope:country = {
    # # Setup/Refresh the list of all building types
    # if = { limit = { NOT = { has_global_variable_list = every_building_type_list } }
    #   lsq_setup_every_building_type_list = {}
    # }

    # Setup the building type type maps
    if = { limit = { OR = { NOT = { has_global_variable_map = lsq_building_type_to_is_production_building } NOT = { has_global_variable_map = lsq_building_type_to_is_non_production_multi_level_building } NOT = { has_global_variable_map = lsq_building_type_to_is_one_per_location_building } NOT = { has_global_variable_map = lsq_building_type_to_is_fort } } }
      lsq_populate_building_type_type_maps = {}
    }

    # Reset the location lists every time we load the game
    if = { limit = { OR = { NOT = { has_variable_list = lsq_all_controlled_locations_list } NOT = { has_variable_list = lsq_all_locations_including_subjects_list } } }
      lsq_reset_locations_lists = {}
    }

    if = { limit = { NOT = { has_variable_list = lsq_auto_build_building_types_list } }
      lsq_reset_auto_build_priorities_to_default_effect = {}
    }
    if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_buildings } }
      set_variable = { name = lsq_minimum_gold_for_buildings value = 50 }
    }
    if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_rgos } }
      set_variable = { name = lsq_minimum_gold_for_rgos value = 50 }
    }
    if = { limit = { NOT = { has_variable = lsq_minimum_gold_for_auto_build } }
      set_variable = { name = lsq_minimum_gold_for_auto_build value = 50 }
    }
    if = { limit = { NOT = { has_variable = lsq_required_rgo_levels_before_sawmill } }
      set_variable = { name = lsq_required_rgo_levels_before_sawmill value = 5 }
    }
    if = { limit = { NOT = { has_variable = lsq_required_rgo_levels_before_windmill } }
      set_variable = { name = lsq_required_rgo_levels_before_windmill value = 5 }
    }
    if = { limit = { NOT = { has_variable = lsq_required_rgo_levels_before_local_smelters } }
      set_variable = { name = lsq_required_rgo_levels_before_local_smelters value = 5 }
    }

    if = { limit = { NOT = { has_variable_map = lsq_minimum_control_for_auto_build } }
      every_in_list = {
        variable = lsq_auto_build_building_types_list
        save_scope_as = building_type
        scope:country = {
          if = { limit = { OR = { NOT = { has_variable_map = lsq_minimum_control_for_auto_build } NOT = { is_key_in_variable_map = { name = lsq_minimum_control_for_auto_build target = scope:building_type } } } }
            # Default to 0 for temple and minting office as they increase control so best to build them everywhere regardless of control
            if = { limit = { OR = { scope:building_type = building_type:temple scope:building_type = building_type:minting_office } }
              add_to_variable_map = { name = lsq_minimum_control_for_auto_build key = scope:building_type value = 0 }
            } else = {
              add_to_variable_map = { name = lsq_minimum_control_for_auto_build key = scope:building_type value = 0.5 }
            }
          }
        }
      }
    }
  }
}

# Expects scopes: country
lsq_reset_locations_lists = {
  scope:country = {
    # Make list of all directly owned locations
    # Also, add to the beginning of the all including subjects list, the sorted list of directly owned locations
    clear_variable_list = lsq_all_controlled_locations_list
    clear_variable_list = lsq_all_locations_including_subjects_list
    ordered_owned_location = {
      order_by = max_control
      save_scope_as = location
      max = lsq_max_guess_at_number_of_locations
      check_range_bounds = no
      scope:country = { add_to_variable_list = { name = lsq_all_controlled_locations_list target = scope:location } }
      scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list target = scope:location } }
      # debug_log = "Adding location [THIS.GetLocation.GetNameWithNoTooltip] to lsq_all_controlled_locations_list with max control [THIS.GetLocation.GetMaxControl]"
    }

    # Make unsorted list of locations of all subjects and below
    clear_variable_list = lsq_all_locations_including_subjects_list_temp
    every_subject_or_below = {
      every_owned_location = {
        save_scope_as = location
        scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list_temp target = scope:location } }
      }
    }

    # Sort and add the subject locations to the end of the list
    ordered_in_list = {
      variable = lsq_all_locations_including_subjects_list_temp
      order_by = max_control
      max = lsq_max_guess_at_number_of_locations
      check_range_bounds = no
      save_scope_as = location
      scope:country = { add_to_variable_list = { name = lsq_all_locations_including_subjects_list target = scope:location } }
    }
    clear_variable_list = lsq_all_locations_including_subjects_list_temp
  }
}
