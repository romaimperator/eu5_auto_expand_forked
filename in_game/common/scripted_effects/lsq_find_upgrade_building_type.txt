# Effect expects building_type scope. Sets the scope return_scope to the upgrade building type, or leaves it unset if not upgradeable.
# Upgrade mappings are populated by the hidden window GUI via BuildingType.GetNextReplaced.
lsq_find_upgrade_building_type = {
	if = { limit = { is_key_in_global_variable_map = { name = lsq_building_type_to_upgrade_type target = this } }
		"global_variable_map(lsq_building_type_to_upgrade_type|this)" = { save_scope_as = return_scope }
	} else = {
		if = { limit = { debug_only = yes } debug_log = "No upgrade building type found" }
	}
}
