on_game_start = {
  on_actions = {
    lsq_on_game_start
  }
}

lsq_on_game_start = {
  effect = {
    debug_log = "LSQ: on_game_start fired"
    # Set the version of the mod that was used to start the game to the global variables
    if = { limit = { lsq_is_version_set = no }
      lsq_set_mod_version_to_current = {}
    }
  }
}
