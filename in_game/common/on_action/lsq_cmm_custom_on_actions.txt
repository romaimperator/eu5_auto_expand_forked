# Custom CMM hooks not managed by the visual editor.
# Runs after the visual-editor-managed registration completes.
cmf_on_mod_registration = {
	on_actions = {
		lsq_on_post_registration
	}
}

lsq_on_post_registration = {
	effect = {
		lsq_cmm_post_registration = yes
	}
}

# Country transfer hook. CMF has already copied CMM setting values.
# Copy LSQ-specific non-CMM data (location lists, etc.).
cmf_on_country_transfer = {
	on_actions = {
		lsq_on_country_transfer
	}
}

lsq_on_country_transfer = {
	effect = {
		# Rebuild location lists for the new country
		save_temporary_scope_as = country
		lsq_reset_locations_lists = {}

		# Copy simple setting aliases from old country
		if = { limit = { scope:old_country = { has_variable = lsq_pause_auto_expand } }
			set_variable = { name = lsq_pause_auto_expand value = scope:old_country.var:lsq_pause_auto_expand }
		}
		if = { limit = { scope:old_country = { has_variable = lsq_should_not_upgrade_buildings } }
			set_variable = { name = lsq_should_not_upgrade_buildings value = scope:old_country.var:lsq_should_not_upgrade_buildings }
		}
		if = { limit = { scope:old_country = { has_variable = lsq_minimum_gold } }
			set_variable = { name = lsq_minimum_gold value = scope:old_country.var:lsq_minimum_gold }
		}
		if = { limit = { scope:old_country = { has_variable = lsq_allowed_building_overbuild_levels } }
			set_variable = { name = lsq_allowed_building_overbuild_levels value = scope:old_country.var:lsq_allowed_building_overbuild_levels }
		}
	}
}
