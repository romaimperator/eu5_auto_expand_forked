cmf_monthly_human_country_pulse = {
	on_actions = {
		lsq_unified_auto_expand
	}
}

# Unified dispatcher: iterates the priority list in order and calls each
# enabled feature's effect. Gated by the master pause toggle.
lsq_unified_auto_expand = {
	effect = {
		save_scope_as = country
		if = { limit = { lsq_should_pause_auto_expand = no }
			if = { limit = { has_variable_list = lsq_priority_features_list }
				every_in_list = {
					variable = lsq_priority_features_list
					save_scope_as = feature
					if = { limit = { scope:country = { is_target_in_variable_list = { name = lsq_priority_features_enabled target = scope:feature } } }
						switch = {
							trigger = scope:feature
							flag:lsq_feature_auto_expand_buildings = { lsq_run_auto_expand_buildings = yes }
							flag:lsq_feature_auto_expand_rgos = { lsq_run_auto_expand_rgos = yes }
							flag:lsq_feature_upgrade_all_buildings = { lsq_run_upgrade_all_buildings = yes }
							flag:lsq_feature_auto_build = { lsq_run_auto_build = yes }
							flag:lsq_feature_auto_food = { lsq_run_auto_food = yes }
							flag:lsq_feature_auto_upgrade_towns_to_cities = { lsq_run_auto_upgrade_towns_to_cities = yes }
						}
					}
				}
			}
		}
	}
}
