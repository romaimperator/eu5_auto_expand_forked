monthly_country_pulse = {
	on_actions = {
		lsq_auto_build_one_per_location_buildings
	}
}

# Auto-build one-per-location buildings (temple, etc.) when enabled in settings.
lsq_auto_build_one_per_location_buildings = {
	trigger = {
		is_ai = no
	}

	effect = {
		save_scope_as = country
		lsq_auto_build_init_locations_lists = {} # Update the lists of locations
    # debug_log = "Auto-building one-per-location buildings"
		every_in_list = {
			variable = lsq_auto_build_building_types_list
			save_scope_as = building_type
			# debug_log = "Checking building type [THIS.GetBuildingType.GetNameWithNoTooltip]"
			scope:country = {
				if = { limit = { is_target_in_variable_list = { name = lsq_auto_build_building_types_enabled target = scope:building_type } }
					# debug_log = "Building [TopScope.sBuildingType('building_type').GetNameWithNoTooltip] is enabled"
					if = { limit = { NOT = { is_target_in_variable_list = { name = lsq_auto_build_building_types_in_subjects_enabled target = scope:building_type } } }
						ordered_controlled_location = {
							order_by = max_control
							max = lsq_max_guess_at_number_of_locations
							check_range_bounds = no
							save_scope_as = location
							# requires a script value to fetch the goods demand in the market
							# debug_log = "Checking location [THIS.GetLocation.GetNameWithNoTooltip] with books demand at [GuiScope.SetRoot(THIS.GetLocation.MakeScope).ScriptValue('lsq_get_good_demand')] and lsq_get_max_available_construction_good_to_avoid_stalling at [GuiScope.SetRoot(THIS.GetLocation.MakeScope).ScriptValue('lsq_get_max_available_construction_good_to_avoid_stalling')]"
							if = { limit = { AND = { lsq_get_open_build_queue_slots > 0 lsq_location_can_auto_build = yes } }
								# debug_log = "Building [TopScope.sBuildingType('building_type').GetNameWithNoTooltip] in [THIS.GetLocation.GetNameWithNoTooltip]"
								construct_building = { building_type = scope:building_type }
							}
						}
					} else = {
						every_in_list = {
							variable = lsq_all_locations_including_subjects_list
							save_scope_as = location
							if = { limit = { AND = { lsq_get_open_build_queue_slots > 0 lsq_location_can_auto_build = yes } }
								if = { limit = { owner = scope:country }
									construct_building = { building_type = scope:building_type }
								} else = {
									# debug_log = "Building [SCOPE.sBuildingType('building_type').GetNameWithNoTooltip] in [THIS.GetLocation.GetNameWithNoTooltip]"
									construct_building = {
										building_type = scope:building_type
										owner = scope:country
										payer = scope:country
									}
								}
							}
						}
					}
				}
			}
		}
