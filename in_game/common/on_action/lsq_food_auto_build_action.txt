# Daily tick: iterate enabled food RGO types in priority order and build optimally.
lsq_food_auto_build_daily_tick = {
	effect = {
		save_scope_as = country
		if = { limit = { lsq_should_pause_auto_expand = no }
			if = { limit = { has_variable_list = lsq_auto_food_rgo_types_list }
				every_in_list = {
					variable = lsq_auto_food_rgo_types_list
					save_scope_as = food_good
					scope:country = {
						if = { limit = { is_target_in_variable_list = { name = lsq_auto_food_rgo_types_enabled target = scope:food_good } }
							lsq_food_auto_build_for_rgo_type = yes
						}
					}
				}
			}
		}
	}
}
