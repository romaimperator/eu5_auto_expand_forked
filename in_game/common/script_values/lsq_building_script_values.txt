# Expects scopes: building_type
lsq_base_building_type_cost = {
	value = scope:building_type.building_base_cost_in_gold
}

# Expects scopes: location, building_type, country
lsq_adjustment_to_base_building_cost = {
	scope:location = {
		value = 0
		scope:country = { add = modifier:global_build_buildings_cost }
		if = { limit = { lsq_does_building_type_count_for_fort_building_cost_modifier = yes }
			scope:country = { add = modifier:global_fort_build_buildings_cost }
		}
		if = { limit = { lsq_does_building_type_count_for_port_building_cost_modifier = yes }
			scope:country = { add = modifier:global_port_build_buildings_cost }
			add = modifier:local_port_build_buildings_cost
		}
		if = { limit = { lsq_auto_build_check_town_or_city = yes }
			scope:country = { add = modifier:global_urban_build_buildings_cost }
		} else_if = { limit = { lsq_auto_build_check_rural = yes }
			scope:country = { add = modifier:global_rural_build_buildings_cost }
		}
		add = modifier:local_build_buildings_cost
		add = modifier:local_build_new_buildings_cost
		if = { limit = { AND = { exists = scope:upgrade_building_type lsq_building_levels_left_to_upgrade > 0 } }
			add = define:NEconomy|UPGRADE_PRICE_MODIFIER
		}
	}
}

# Root is the location, also expects scopes: building_type, country, and upgrade_building_type
lsq_building_levels_left_to_upgrade = {
	value = 0
	"building(scope:building_type|scope:country)" = { add = building_level }
	every_buildings_in_location = {
		limit = { building_type = scope:upgrade_building_type }
		subtract = building_levels_under_construction
	}
}

# Root is the location, also expects scope:building_type, scope:country
lsq_adjusted_building_type_cost = {
	if = { limit = { NOT = { exists = scope:location } }
		save_temporary_scope_as = location
	}
	if = { limit = { exists = scope:upgrade_building_type }
		scope:upgrade_building_type = { value = building_base_cost_in_gold }
	} else = {
		scope:building_type = { value = building_base_cost_in_gold }
	}
	multiply = {
		value = 1
		add = lsq_adjustment_to_base_building_cost
	}
	multiply = {
		value = 1
		add = lsq_get_construction_cost_adjustment_for_building_type
	}
	multiply = {
		value = 1
		add = scope:country.inflation
	}
}

# Expects scopes: building_type, country, and location
lsq_adjusted_building_type_cost_accounting_for_upgrades = {
	scope:location = {
		if = { limit = { scope:building_type = { lsq_is_building_type_obsolete = no } }
			value = lsq_adjusted_building_type_cost
		} else_if = { limit = { lsq_is_building_type_upgradeable_and_can_be_upgraded_by_owner = yes }
			value = lsq_adjusted_upgrade_cost
		} else = { value = -1 } # should never happen
	}
}

# Root is the building type, also expects scope:country
lsq_double_building_type_cost = {
	if = { limit = { lsq_is_building_type_obsolete = no }
		value = building_base_cost_in_gold
	} else_if = { limit = { lsq_is_building_type_upgradeable_and_can_be_upgraded_by_owner = yes }
		value = lsq_get_upgrade_building_cost
	} else = { value = 100 } # should never happen
	multiply = 2
}

# Root is the building type, also expects scope:country and scope:location
lsq_expected_gold_available_for_building_type_for_auto_expand = {
	if = { limit = { NOT = { exists = scope:building_type } }
		save_temporary_scope_as = building_type
	}
	scope:location = { value = lsq_adjusted_building_type_cost_accounting_for_upgrades }
	scope:country = {

# TODO: Currently a pessimistic calculation to avoid stalling but will improve in a future update.
# Game doesn't seem to stall a building even when the supply is technically insufficient as long as part of the amount
# required is available. Using this until we can verify that and see if the game slows down construction when the supply
# is insufficient.
# Root is the location, also expects scope:good
lsq_get_max_available_construction_good_to_avoid_stalling = {
	save_temporary_value_as = {
		name = stockpile_minus_supply
		value = {
			value = "market.stockpile_in_market(scope:good)"
			subtract = "market.goods_supply_in_market(scope:good)"
		}
	}
	if = { limit = { scope:stockpile_minus_supply < 0.05 } # assume there's no stockpile if the supply and stockpile are basically equal
		value = "market.goods_supply_in_market(scope:good)"
		subtract = "market.goods_demand_in_market(scope:good)"
	} else = {
		value = "market.goods_supply_in_market(scope:good)"
		add = {
			value = "market.stockpile_in_market(scope:good)"
			divide = 12 # require 12 times as much stockpile compared to supply to avoid stalling. Real calc would consider how long the demand would be for.
		}
		subtract = "market.goods_demand_in_market(scope:good)"
	}
}

# Expects scopes: good, location
lsq_get_construction_good_adjustment = {
	if = { limit = { "scope:location.market.market_price(scope:good)" > "default_price(scope:good)" }
		value = "scope:location.market.market_price(scope:good)"
		divide = "default_price(scope:good)"
		subtract = 1
		divide = 3
	} else_if = { limit = { "scope:location.market.market_price(scope:good)" < "default_price(scope:good)" }
		value = "default_price(scope:good)"
		divide = "scope:location.market.market_price(scope:good)"
		subtract = 1
		multiply = -10
	} else = {
		value = 0
	}
}
