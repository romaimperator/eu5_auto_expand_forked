lsq_double_building_cost = {
	if = { limit = { lsq_is_building_obsolete = no }
		building_type = { value = building_base_cost_in_gold }
	} else_if = { limit = { lsq_is_building_upgradeable_and_can_be_upgraded_by_owner = yes }
		building_type = { value = lsq_get_upgrade_building_cost }
	} else = { value = 100 } # should never happen
	multiply = 2
}

lsq_expected_gold_available_for_building = {
	value = lsq_double_building_cost
	if = { limit = { exists = location.owner.var:lsq_minimum_gold_for_buildings }
		add = location.owner.var:lsq_minimum_gold_for_buildings
	}
}

lsq_get_minimum_gold_for_buildings = {
	value = var:lsq_minimum_gold_for_buildings
}
