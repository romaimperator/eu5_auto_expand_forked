# Root is the building_type. Expects scopes: location.
# Wrapper that ensures scope:building_type is set, then delegates to the full calculation.
lsq_get_construction_cost_adjustment_for_building_type = {
	if = { limit = { NOT = { exists = scope:building_type } } save_temporary_scope_as = building_type }
	value = lsq_get_construction_cost_modifier_for_building_type
}

# Standard-goods-only wrapper (lumber, masonry, glass, sand, stone). Used for the min
# discount threshold so that non-depressible goods (gold, tools, paper, etc.) don't block building.
lsq_get_standard_construction_cost_adjustment_for_building_type = {
	if = { limit = { NOT = { exists = scope:building_type } } save_temporary_scope_as = building_type }
	value = lsq_get_standard_construction_cost_modifier_for_building_type
}

# Cost modifier calculation using per-good construction demand maps.
# These maps are populated by the hidden window GUI for all building types.
# The 28 construction goods below are a stable game constant.
#
# Algorithm: for each construction good, clamp the raw price adjustment to [-0.33, 0.33],
# then weight by demand share (demand_for_good / total_demand). Since weights sum to 1.0,
# the result naturally stays in [-0.33, 0.33] without further normalization.
#
# Expects scopes: location, building_type

      goods:books = { save_temporary_scope_as = good }
      value = lsq_get_construction_good_adjustment
