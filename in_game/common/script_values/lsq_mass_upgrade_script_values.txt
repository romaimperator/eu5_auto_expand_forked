# Root expects scope:country. Returns the minimum gold the country must have before queueing a city upgrade.
# Equal to the base upgrade cost (2000) plus the shared gold reserve setting.
lsq_mass_upgrade_town_to_city_gold_needed = {
	value = 2000
	scope:country = { add = lsq_get_minimum_gold }
}

# Root expects scope:country. Returns the minimum gold the country must have before queueing a town upgrade.
# Equal to the base upgrade cost (500) plus the shared gold reserve setting.
lsq_mass_upgrade_rural_to_town_gold_needed = {
	value = 500
	scope:country = { add = lsq_get_minimum_gold }
}

# Returns the RGO base price threshold above which rural-to-town promotion is skipped.
# Reads the player-configured slider value from the country alias variable.
lsq_get_rural_to_town_max_rgo_price = {
	value = scope:country.var:lsq_rural_to_town_max_rgo_price
}
