# Per-level cost multipliers for buildings with increase_per_level_cost.
# Only a few buildings have this property — maintain manually if the game adds more.
# Root is the building_type
# Expects scopes: location
lsq_get_increase_per_level_cost_for_building_type = {
	if = { limit = { this = building_type:construction_center } value = 0.5 }
	else_if = { limit = { this = building_type:fondaco } value = 0.5 }
	else_if = { limit = { this = building_type:hanseatic_kontor } value = 0.5 }
	else_if = { limit = { this = building_type:hanseatic_town_hall } value = 0.5 }
	else_if = { limit = { this = building_type:seljuk_mint } value = 0.1 }
	else = { value = 0 }
  scope:location = { multiply = { value = 1 add = "location_building_level(prev)" } }
}
