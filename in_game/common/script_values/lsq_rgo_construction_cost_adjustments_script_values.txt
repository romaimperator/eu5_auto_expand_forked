# RGO construction cost adjustments based on primary construction good per RGO type.
# Only 5 RGO types exist — maintain manually if the game adds more.
# Expects scopes: country, location
lsq_get_construction_cost_adjustment_for_rgo = {
	if = { limit = { scope:location.raw_material = { goods_method = mining } }
		value = lsq_get_construction_cost_adjustment_for_rgo_mining
	}
	else_if = { limit = { scope:location.raw_material = { goods_method = farming } }
		value = lsq_get_construction_cost_adjustment_for_rgo_farming
	}
	else_if = { limit = { scope:location.raw_material = { goods_method = hunting } }
		value = lsq_get_construction_cost_adjustment_for_rgo_hunting
	}
	else_if = { limit = { scope:location.raw_material = { goods_method = gathering } }
		value = lsq_get_construction_cost_adjustment_for_rgo_gathering
	}
	else_if = { limit = { scope:location.raw_material = { goods_method = forestry } }
		value = lsq_get_construction_cost_adjustment_for_rgo_forestry
	}
	else = {
		value = 0
	}
}

# Expects scopes: country, location
lsq_get_construction_cost_adjustment_for_rgo_mining = {
	value = 0
	add = {
		goods:lumber = { save_temporary_scope_as = good }
		value = lsq_get_construction_good_adjustment
		min = -0.33
		max = 0.33
	}
}
