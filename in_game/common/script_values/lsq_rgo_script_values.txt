# max cost is 50 gold base times an additional 33%. Using 50*1.34 = 67 for extra buffer.
# lsq_expand_rgo_max_cost = 67
lsq_expand_rgo_max_cost = 100 # Changing to 100 to align it with first level building costs of double 50 gold

# Root scope is the player (passed from GUI)
lsq_expected_gold_available_for_rgo = {
	value = lsq_expand_rgo_max_cost
	if = { limit = { exists = owner.var:lsq_minimum_gold_for_rgos }
		add = owner.var:lsq_minimum_gold_for_rgos
	}
}

lsq_get_minimum_gold_for_rgos = {
	value = var:lsq_minimum_gold_for_rgos
}

lsq_get_max_rgo_workers = {
	value = max_rgo_workers
}

lsq_get_max_rgo_workers_floor = {
	value = max_rgo_workers
	floor = yes
}

lsq_get_rgo_workers = {
	value = rgo_workers
}

lsq_get_ui_saved_max_rgo_workers = {
	value = var:lsq_max_rgo_workers
}

lsq_get_num_civil_constructions = {
	value = num_civil_constructions
}

lsq_get_buildings_under_construction = {
	value = 0
	every_buildings_in_location = {
		add = building_levels_under_construction
	}
}

lsq_get_rgos_under_construction = {
	value = num_civil_constructions
	subtract = lsq_get_buildings_under_construction
}

lsq_get_max_constructions_at_same_time = {
	value = modifier:max_constructions_at_same_time
}

# Root scope is the location
lsq_get_open_build_queue_slots = {
	value = lsq_get_max_constructions_at_same_time
	subtract = num_civil_constructions
}

# Root scope is the location
lsq_modifier_rgo_level = {
  value = modifier:lsq_rgo_level
}
