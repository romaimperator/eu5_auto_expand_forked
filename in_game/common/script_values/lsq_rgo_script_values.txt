# 50 gold is the base cost of all types of RGO defined in in_game/common/prices/00_hardcoded.txt
lsq_get_defined_base_cost_for_rgo = 50

# Expects scopes: country, location
lsq_expected_gold_available_for_rgo_auto_expand = {
	value = lsq_get_adjusted_construction_cost_for_rgo
	if = { limit = { exists = scope:country.var:lsq_minimum_gold }
		add = scope:country.var:lsq_minimum_gold
	}
}

# Expects scopes: country, location. Sum of all multiplicative-style adjustments to the base RGO cost.
lsq_get_adjustment_to_base_cost_for_rgo = {
	value = 0
	add = {
		# Per-good base cost adjustment: build cost scales with the good's default price; expensive goods cost more, cheap goods less.
		value = "default_price(scope:location.raw_material)"
		subtract = 2
		divide = 4
	}
	if = { limit = { scope:location.raw_material = { goods_method = mining } }
		add = scope:country.modifier:expand_rgo_mining_cost_modifier
	}
	if = { limit = { scope:location.raw_material = { goods_method = farming } }
		add = scope:country.modifier:expand_rgo_farming_cost_modifier
	}
	if = { limit = { scope:location.raw_material = { goods_method = hunting } }
		add = scope:country.modifier:expand_rgo_hunting_cost_modifier
	}
	if = { limit = { scope:location.raw_material = { goods_method = gathering } }
		add = scope:country.modifier:expand_rgo_gathering_cost_modifier
	}
	if = { limit = { scope:location.raw_material = { goods_method = forestry } }
		add = scope:country.modifier:expand_rgo_forestry_cost_modifier
	}
	add = scope:country.inflation
}

# Expects scopes: country, location
lsq_get_adjusted_construction_cost_for_rgo = {
	value = lsq_get_defined_base_cost_for_rgo
	multiply = {
		value = 1
		add = lsq_get_adjustment_to_base_cost_for_rgo
	}
	multiply = {
		value = 1
		add = lsq_get_construction_cost_adjustment_for_rgo
	}
}

lsq_get_max_rgo_workers = {
	value = max_rgo_workers
}

lsq_get_max_rgo_workers_floor = {
	value = max_rgo_workers
	floor = yes
}

lsq_get_rgo_workers = {
	value = rgo_workers
}

lsq_get_num_civil_constructions = {
	value = num_civil_constructions
}

lsq_get_buildings_under_construction = {
	value = 0
	every_buildings_in_location = {
		add = building_levels_under_construction
	}
}

lsq_get_rgos_under_construction = {
	value = num_civil_constructions
	subtract = lsq_get_buildings_under_construction
}

lsq_get_max_constructions_at_same_time = {
	value = modifier:max_constructions_at_same_time
}

# Root scope is the location
lsq_get_open_build_queue_slots = {
	value = lsq_get_max_constructions_at_same_time
	subtract = num_civil_constructions
}

# Root scope is the location
lsq_modifier_rgo_level = {
  value = modifier:lsq_rgo_level
}

# Root is the location. Estimates gold profit per RGO level.
# Uses total raw material output / current levels × market price as a proxy.
lsq_rgo_estimated_profit_per_level = {
	raw_material = { save_temporary_scope_as = rgo_good }
	value = raw_material_output
	if = { limit = { lsq_modifier_rgo_level >= 1 }
		divide = lsq_modifier_rgo_level
	}
	multiply = "market.market_price(scope:rgo_good)"
}
