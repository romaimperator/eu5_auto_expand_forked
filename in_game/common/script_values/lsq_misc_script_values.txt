# Root is country. Expects scope building_type. Returns required RGO/lumber levels before auto-building this improvement (sawmill/windmill/local_smelters).
lsq_get_required_rgo_levels_before_improvement_building = {
	if = { limit = { scope:building_type = building_type:sawmill }
		value = var:lsq_required_rgo_levels_before_sawmill
	} else_if = { limit = { scope:building_type = building_type:windmill }
		value = var:lsq_required_rgo_levels_before_windmill
	} else_if = { limit = { scope:building_type = building_type:local_smelters }
		value = var:lsq_required_rgo_levels_before_local_smelters
	} else = { value = 0 }
}

# Root is the country
lsq_get_allowed_building_overbuild_levels = {
	if = { limit = { exists = var:lsq_allowed_building_overbuild_levels }
		value = var:lsq_allowed_building_overbuild_levels
	} else = { value = 0 }
}

}

# Includes potential employment from levels under construction and from buildings that are subsidized but not fully employed
# Expects scope: building
lsq_building_employment_required = {
  value = 0
  scope:building = {
    if = { limit = { is_opened = yes }
      if = { limit = { AND = { building_type = { lsq_is_building_type_production_building = yes } is_subsidized = yes } }
        add = {
          value = building_level
          multiply = building_type.employment_size
        }
      } else = {
        add = building_employment_size_amount
      }
      add = {
        value = building_levels_under_construction
        multiply = building_type.employment_size
      }
    }
  }
}

lsq_location_lumber_mill_level = {
  if = { limit = { has_building = building_type:lumber_mill }
    "building(building_type:lumber_mill|owner)" = { value = building_level }
  } else = {
    value = -1
  }
}
