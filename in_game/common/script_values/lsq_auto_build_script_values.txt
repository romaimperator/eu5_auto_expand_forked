lsq_get_minimum_control_for_auto_build = {
  scope:country = {
    if = { limit = { AND = { has_variable_map = lsq_min_control_by_building_type is_key_in_variable_map = { name = lsq_min_control_by_building_type target = prev } } }
      value = "variable_map(lsq_min_control_by_building_type|prev)"
    } else = { value = -1 }
  }
}
