# Directly controlled #
#######################

# Root is the country. Also expects the scope building_type and country.
lsq_auto_build_number_of_locations_allowed_to_build = {
  value = 0
  if = { limit = { scope:building_type = building_type:temple }
    every_controlled_location = {
      limit = { lsq_location_allowed_to_build_temple = yes }
      add = 1
    }
  } else_if = { limit = { scope:building_type = building_type:library }
    every_controlled_location = {
      limit = { lsq_location_allowed_to_build_library = yes }
      add = 1
    }
  } else_if = { limit = { scope:building_type = building_type:settlement_building }
    every_controlled_location = {
      limit = { lsq_location_allowed_to_build_settlement_building = yes }
      add = 1
    }
  } else_if = { limit = { scope:building_type = building_type:bridge_infrastructure }
    every_controlled_location = {
      limit = { lsq_location_allowed_to_build_bridge_infrastructure = yes }
      add = 1
    }
  } else_if = { limit = { scope:building_type = building_type:university }
    every_controlled_location = {
      limit = { lsq_location_allowed_to_build_university = yes }
      add = 1
    }
  }
}
