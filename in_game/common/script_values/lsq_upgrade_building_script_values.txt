# Upgrade building cost/adjustment script values.
# Upgrade mappings are populated by the hidden window GUI via BuildingType.GetNextReplaced.

# Script value expects building_type scope. Returns the base gold cost of the upgrade building type.
lsq_get_upgrade_building_cost = {
	if = { limit = { is_key_in_global_variable_map = { name = lsq_building_type_to_upgrade_type target = this } }
		"global_variable_map(lsq_building_type_to_upgrade_type|this)" = { value = building_base_cost_in_gold }
	}
}

# Script value expects scope:building_type. Returns the adjusted upgrade cost including location modifiers.
lsq_adjusted_upgrade_cost = {
	if = { limit = { is_key_in_global_variable_map = { name = lsq_building_type_to_upgrade_type target = scope:building_type } }
		"global_variable_map(lsq_building_type_to_upgrade_type|scope:building_type)" = { save_temporary_scope_as = upgrade_building_type }
		value = lsq_adjusted_building_type_cost
	}
}

# Script value expects scope:building_type. Returns the cost adjustment for upgrading.
lsq_get_adjustment_to_base_building_cost_for_upgrade = {
	if = { limit = { is_key_in_global_variable_map = { name = lsq_building_type_to_upgrade_type target = scope:building_type } }
		"global_variable_map(lsq_building_type_to_upgrade_type|scope:building_type)" = { save_temporary_scope_as = upgrade_building_type }
		value = lsq_adjustment_to_base_building_cost
	}
}
