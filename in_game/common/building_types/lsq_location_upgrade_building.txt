lsq_upgrade_town_to_city = {
	is_foreign = no
	pop_type = laborers
	max_levels = 1

	category = infrastructure_category
	employment_size = rural_peasant_produce_employment

	town = yes
	city = yes
	price = lsq_city_upgrade_price

	forbidden_for_estates = yes

	country_potential = {
		is_ai = no
	}
	build_time = 730
	construction_demand = build_city_demand
	allow = {
		location_rank = location_rank:town
		population >= 30
	}
	location_potential = {
		modifier:cannot_upgrade_location = no
	}

	unique_production_methods = {
		lsq_upgrade_town_to_city_maintenance = {
			masonry = 0.01
			category = building_maintenance
		}
	}

	modifier = {
	}

	on_built = {
		location = {
			if = {
				limit = {
					location_rank = location_rank:town
				}
				change_location_rank = location_rank:city
			}
			trigger_event_silently = {
				id = lsq_location_upgrade.1
				days = 1
			}
		}
	}
}
