building_only_auto_expand = {
	scope = building
	tag = building
	group = 9
	exclusive_group = yes
	trigger = {
		OR = {
			location = { is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = prev.building_type } }
			AND = {
				owner = { exists = var:lsq_mass_auto_expand_buildings }
				NOT = { location = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = prev.building_type } } }
				building_type = {
					OR = {
						lsq_is_building_type_production_building = yes
						lsq_is_building_type_non_production_multi_level_building = yes
					}
				}
			}
		}
	}
}

building_only_not_auto_expand = {
	scope = building
	tag = building
	group = 9
	exclusive_group = yes
	trigger = {
		NOR = {
			location = { is_target_in_variable_list = { name = lsq_auto_expand_registered_building_types target = prev.building_type } }
			AND = {
				owner = { exists = var:lsq_mass_auto_expand_buildings }
				NOT = { location = { is_target_in_variable_list = { name = lsq_auto_expand_excluded_building_types target = prev.building_type } } }
				building_type = {
					OR = {
						lsq_is_building_type_production_building = yes
						lsq_is_building_type_non_production_multi_level_building = yes
					}
				}
			}
		}
	}
}
